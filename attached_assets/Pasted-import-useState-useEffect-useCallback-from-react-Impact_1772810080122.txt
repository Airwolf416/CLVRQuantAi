// ─── Surprise indicator ────────────────────────────────────────────────────
function getSurprise(event) {
if (!event.actual || !event.forecast) return null;
const a = parseFloat(event.actual.replace(/[^0-9.-]/g, “”));
const f = parseFloat(event.forecast.replace(/[^0-9.-]/g, “”));
if (isNaN(a) || isNaN(f)) return null;
const diff = a - f;
if (Math.abs(diff) < 0.05) return { label: “IN LINE”, color: “#888” };
return diff > 0
? { label: `BEAT +${Math.abs(diff).toFixed(1)}`, color: “#4ade80” }
: { label: `MISS -${Math.abs(diff).toFixed(1)}`, color: “#f87171” };
}

// ─── QuantBrain market impact assessment ──────────────────────────────────
function getMarketImpact(event) {
if (!event.released || !event.actual) return null;
const impacts = [];
const name = event.event.toLowerCase();

if (name.includes(“non-farm”) || name.includes(“nfp”)) {
const actual = parseFloat(event.actual.replace(/[^0-9.-]/g, “”));
if (actual < 0) {
impacts.push({ asset: “USD”, dir: “🔴 BEARISH”, reason: “Jobs collapse → rate cut expectations surge” });
impacts.push({ asset: “XAU”, dir: “🟢 BULLISH”, reason: “Safe haven demand + rate cut tailwind” });
impacts.push({ asset: “BTC”, dir: “🟡 CAUTIOUS”, reason: “Risk-off initially but rate cuts medium-term bullish” });
impacts.push({ asset: “SPX”, dir: “🔴 BEARISH”, reason: “Recession fears dominate” });
}
}
if (name.includes(“gdp”)) {
impacts.push({ asset: “EUR”, dir: “🔴 BEARISH”, reason: “Below forecast GDP → ECB cut odds rise” });
impacts.push({ asset: “EUR/USD”, dir: “🔴 BEARISH”, reason: “Euro weakness on slow growth” });
}
if (name.includes(“earnings”) || name.includes(“wages”)) {
impacts.push({ asset: “USD”, dir: “🟡 MIXED”, reason: “Hot wages = sticky inflation, limits Fed cuts” });
}
return impacts.length > 0 ? impacts : null;
}

function AIModal({ event, onClose }) {

const [loading, setLoading] = useState(true);

useEffect(() => {
const ask = async () => {
try {
const res = await fetch(“https://api.anthropic.com/v1/messages”, {
method: “POST”,
headers: { “Content-Type”: “application/json” },
body: JSON.stringify({
model: “claude-sonnet-4-20250514”,
max_tokens: 1000,
system: `You are QuantBrain, an elite quantitative market intelligence analyst for CLVRQuant AI. You provide concise, data-driven analysis of economic releases. Focus on: 1) What the data means for markets, 2) Which assets are most affected, 3) How this changes the macro picture, 4) What to watch next. Always remind users this is information only, not financial advice. Be precise and use numbers.`,
messages: [{
role: “user”,
content: `Analyze this economic release for me:

Event: ${event.event}
Country/Region: ${event.region}
Forecast: ${event.forecast} ${event.unit}
Previous: ${event.previous} ${event.unit}
Actual: ${event.actual || “Not yet released”} ${event.unit}
Impact Level: ${event.impact.toUpperCase()}
Today’s Date: March 6, 2026

${event.actual ? `The actual came in ${parseFloat(event.actual) > parseFloat(event.forecast) ? "ABOVE" : "BELOW"} expectations.` : “This event has not yet been released.”}

What does this mean for markets? Which assets move? What’s the macro implication? What should I watch next?`
}]
})
});
const data = await res.json();
setResponse(data.content?.map(c => c.text || “”).join(”\n”) || “No response.”);
} catch (e) {
setResponse(“❌ Error: “ + e.message);

setLoading(false);

ask();
