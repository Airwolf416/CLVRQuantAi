function intervalToMs(interval) {
const map = { “1m”:60000, “5m”:300000, “15m”:900000, “1h”:3600000, “4h”:14400000, “1d”:86400000 };
return map[interval] || 3600000;
}

// PATH A: Hyperliquid — crypto perps (free, no key)
async function fetchHLCandles(ticker, interval, count) {
try {
const endTime   = Date.now();
const startTime = endTime - (count * intervalToMs(interval));
const r = await fetch(“https://api.hyperliquid.xyz/info”, {
method:  “POST”,
headers: { “Content-Type”: “application/json” },
body: JSON.stringify({
type: “candleSnapshot”,
req:  { coin:ticker, interval, startTime, endTime },
}),
});
const data = await r.json();
if (!Array.isArray(data) || data.length === 0) return null;
return data.map(c => ({
t:parseFloat(c.t), o:parseFloat(c.o), h:parseFloat(c.h),
l:parseFloat(c.l), c:parseFloat(c.c), v:parseFloat(c.v||0),
}));
} catch (e) {
console.error(`[HL candles] ${ticker}:`, e.message);
return null;
}
}

// PATH B: Binance — deeper crypto history (free, no key)
async function fetchBinanceCandles(ticker, interval, count) {
const symbol = BINANCE_SYMBOLS[ticker];
if (!symbol) return null;
try {
const r = await fetch(
`https://api.binance.com/api/v3/klines?symbol=${symbol}&interval=${interval}&limit=${Math.min(count,1000)}`
);
const data = await r.json();
if (!Array.isArray(data)) return null;
return data.map(c => ({
t:parseFloat(c[0]), o:parseFloat(c[1]), h:parseFloat(c[2]),
l:parseFloat(c[3]), c:parseFloat(c[4]), v:parseFloat(c[5]),
}));
} catch (e) {
console.error(`[Binance candles] ${ticker}:`, e.message);
return null;
}
}

// PATH C: Finnhub — equities & commodities (uses FINNHUB_API_KEY secret)
async function fetchFinnhubCandles(ticker, interval, count) {
const apiKey     = process.env.FINNHUB_API_KEY || process.env.REACT_APP_FINNHUB_KEY;
if (!apiKey) return null;
try {
const resMap = { “15m”:“15”,“1h”:“60”,“4h”:“240”,“1d”:“D” };
const res    = resMap[interval] || “60”;
const to     = Math.floor(Date.now() / 1000);
const from   = to - (count * intervalToMs(interval) / 1000);
const r = await fetch(
`https://finnhub.io/api/v1/stock/candle?symbol=${ticker}&resolution=${res}&from=${from}&to=${to}&token=${apiKey}`
);
const data = await r.json();
if (data.s !== “ok” || !data.c?.length) return null;
return data.t.map((t, i) => ({
t:t*1000, o:parseFloat(data.o[i]), h:parseFloat(data.h[i]),
l:parseFloat(data.l[i]), c:parseFloat(data.c[i]), v:parseFloat(data.v?.[i]||0),
}));
} catch (e) {
console.error(`[Finnhub candles] ${ticker}:`, e.message);
return null;
}
}

// Smart fetch — tries best source for asset class, falls back automatically
async function fetchCandles(ticker, assetClass, interval, count) {
if (assetClass === “crypto”) {
// Try HL first, then Binance for deeper history
const [hl, binance] = await Promise.all([
fetchHLCandles(ticker, interval, count),
fetchBinanceCandles(ticker, interval, Math.min(count * 2, 1000)),
]);
// Return longer dataset (more history = better indicators)
if (hl && binance) return binance.length > hl.length ? binance : hl;
return hl || binance;
}
// Equities & commodities — Finnhub first, HL perp fallback
const finnhub = await fetchFinnhubCandles(ticker, interval, count);
if (finnhub) return finnhub;
return fetchHLCandles(ticker, interval, count);
}

// INDICATOR ENGINE — all computed server-side from OHLCV

function computeIndicators(candles) {
if (!candles || candles.length < 50) return null;

const closes  = candles.map(c => c.c);
const highs   = candles.map(c => c.h);
const lows    = candles.map(c => c.l);
const volumes = candles.map(c => c.v);
const n       = closes.length;

// EMA helper
function ema(data, period) {
if (data.length < period) return [data[data.length-1]];
const k   = 2 / (period + 1);
let val   = data.slice(0, period).reduce((a,b)=>a+b,0) / period;
const out = [val];
for (let i = period; i < data.length; i++) {
val = data[i]*k + val*(1-k);
out.push(val);
}
return out;
}

const ema20arr  = ema(closes, 20);
const ema50arr  = ema(closes, 50);
const ema200arr = ema(closes, Math.min(200, n-1));
const ema12arr  = ema(closes, 12);
const ema26arr  = ema(closes, 26);

const currentEma20  = ema20arr[ema20arr.length-1];
const currentEma50  = ema50arr[ema50arr.length-1];
const currentEma200 = ema200arr[ema200arr.length-1];
const currentPrice  = closes[n-1];

// RSI(14)
let gains=0, losses=0;
for (let i=n-14; i<n; i++) {
const diff = closes[i]-closes[i-1];
if (diff>0) gains+=diff; else losses-=diff;
}
const rsi = Math.round(100-100/(1+(gains/Math.max(losses,0.0001))));

// MACD(12,26,9)
const macdLine   = ema12arr.slice(ema12arr.length-ema26arr.length).map((v,i)=>v-ema26arr[i]);
const signalLine = ema(macdLine, 9);
const macd       = macdLine[macdLine.length-1];
const macdSig    = signalLine[signalLine.length-1];
const macdHist   = macd - macdSig;
const prevHist   = macdLine[macdLine.length-2] - signalLine[signalLine.length-2];

// ATR(14)
const trVals = [];
for (let i=Math.max(1,n-30); i<n; i++) {
trVals.push(Math.max(
highs[i]-lows[i],
Math.abs(highs[i]-closes[i-1]),
Math.abs(lows[i]-closes[i-1]),
));
}
const atr14  = trVals.slice(-14).reduce((a,b)=>a+b,0)/14;
const atrPct = (atr14/currentPrice)*100;

// Support & Resistance
const recent50     = candles.slice(-50);
const recentHighs  = recent50.map(c=>c.h).sort((a,b)=>b-a);
const recentLows   = recent50.map(c=>c.l).sort((a,b)=>a-b);
const resistance   = recentHighs.filter(v=>v>currentPrice).slice(0,3);
const support      = recentLows.filter(v=>v<currentPrice).slice(0,3);

// Volume
const avgVol  = volumes.slice(-20).reduce((a,b)=>a+b,0)/20;
const lastVol = volumes[n-1];
const volRatio = lastVol/Math.max(avgVol,0.0001);
const volumeSignal = volRatio>2?“SURGE”:volRatio>1.3?“ABOVE AVG”:volRatio<0.7?“BELOW AVG”:“NORMAL”;

// Trend structure
const trend =
currentPrice>currentEma20 && currentEma20>currentEma50 && currentEma50>currentEma200 ? “STRONG UPTREND” :
currentPrice<currentEma20 && currentEma20<currentEma50 && currentEma50<currentEma200 ? “STRONG DOWNTREND” :
currentPrice>currentEma50 ? “UPTREND” :
currentPrice<currentEma50 ? “DOWNTREND” : “RANGING”;

// Ranges
const high24  = Math.max(…candles.slice(-24).map(c=>c.h));
const low24   = Math.min(…candles.slice(-24).map(c=>c.l));
const high7d  = Math.max(…candles.slice(-Math.min(168,n)).map(c=>c.h));
const low7d   = Math.min(…candles.slice(-Math.min(168,n)).map(c=>c.l));
const posInRange = Math.round((currentPrice-low24)/Math.max(high24-low24,0.0001)*100);

// Momentum composite
let momentum = 50;
if (rsi>60) momentum+=10; if (rsi<40) momentum-=10;
if (macdHist>0&&macdHist>prevHist) momentum+=10; if (macdHist<0&&macdHist<prevHist) momentum-=10;
if (currentPrice>currentEma20) momentum+=8; if (currentPrice<currentEma20) momentum-=8;
if (volumeSignal===“SURGE”) momentum+=5;
momentum = Math.max(0,Math.min(100,momentum));

return {
currentPrice,
ema20:currentEma20, ema50:currentEma50, ema200:currentEma200,
priceVsEma20:((currentPrice-currentEma20)/currentEma20*100).toFixed(2),
priceVsEma50:((currentPrice-currentEma50)/currentEma50*100).toFixed(2),
priceVsEma200:((currentPrice-currentEma200)/currentEma200*100).toFixed(2),
rsi, rsiLabel:rsi>70?“OVERBOUGHT”:rsi>60?“BULLISH”:rsi<30?“OVERSOLD”:rsi<40?“BEARISH”:“NEUTRAL”,
macd:parseFloat(macd.toFixed(4)), macdSignal:parseFloat(macdSig.toFixed(4)),
macdHist:parseFloat(macdHist.toFixed(4)),
macdCrossing: macdHist>0&&prevHist<=0?“BULLISH_CROSS”: macdHist<0&&prevHist>=0?“BEARISH_CROSS”: macdHist>0?“BULLISH”:“BEARISH”,
atr14:parseFloat(atr14.toFixed(6)), atrPct:parseFloat(atrPct.toFixed(3)),
nearestResistance:resistance[0]||null, resistanceLevels:resistance,
nearestSupport:support[0]||null, supportLevels:support,
volumeSignal, volumeRatio:parseFloat(volRatio.toFixed(2)),
trend, momentumScore:momentum,
high24, low24, high7d, low7d, posInRange,
range24hPct:parseFloat(((high24-low24)/low24*100).toFixed(2)),
};
}
