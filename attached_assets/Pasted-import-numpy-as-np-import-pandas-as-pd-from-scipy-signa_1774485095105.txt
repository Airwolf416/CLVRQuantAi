import numpy as np
import pandas as pd
from scipy.signal import argrelextrema
from typing import List, Dict

class PatternRecognitionEngine:
    def __init__(self, window_size: int = 5, variance_tolerance: float = 0.03):
        # Window size determines how wide a peak needs to be to count as a pivot
        self.window = window_size
        # How much asymmetry we allow (e.g., 3% difference between shoulders)
        self.tolerance = variance_tolerance

    def find_pivots(self, df: pd.DataFrame) -> Tuple[np.ndarray, np.ndarray]:
        """
        Uses SciPy to find the mathematical local maxima (peaks) and minima (troughs).
        """
        # Convert close prices to a numpy array for speed
        prices = df['close'].values
        
        # Find indices of peaks and troughs
        peaks_idx = argrelextrema(prices, np.greater, order=self.window)[0]
        troughs_idx = argrelextrema(prices, np.less, order=self.window)[0]
        
        return peaks_idx, troughs_idx

    def detect_head_and_shoulders(self, df: pd.DataFrame) -> bool:
        """
        Scans recent price action for a Bearish Head and Shoulders reversal.
        """
        peaks_idx, _ = self.find_pivots(df)
        prices = df['close'].values
        
        # We need at least 3 peaks to form a Head and Shoulders
        if len(peaks_idx) < 3:
            return False
            
        # Look at the 3 most recent peaks
        recent_peaks = peaks_idx[-3:]
        p1, p2, p3 = prices[recent_peaks[0]], prices[recent_peaks[1]], prices[recent_peaks[2]]
        
        # Geometric Proof:
        # 1. P2 (Head) must be strictly greater than P1 and P3
        is_head_highest = (p2 > p1) and (p2 > p3)
        
        # 2. P1 and P3 (Shoulders) must be relatively equal in height
        shoulder_diff = abs(p1 - p3) / max(p1, p3)
        are_shoulders_symmetrical = shoulder_diff <= self.tolerance
        
        if is_head_highest and are_shoulders_symmetrical:
            print(f"📉 PATTERN DETECTED: Head & Shoulders. Shoulders at {p1:.2f}/{p3:.2f}, Head at {p2:.2f}")
            return True
            
        return False

    def detect_bull_flag(self, df: pd.DataFrame) -> bool:
        """
        Scans for a bullish continuation (Bull Flag).
        A sharp move up (flagpole), followed by a slight downward channel (flag).
        """
        # A simplified mathematical representation for the flagpole:
        # Has the price increased by > 5% in the last 10 periods?
        recent_data = df.tail(15)
        if recent_data.empty:
            return False
            
        start_price = recent_data['close'].iloc[0]
        peak_price = recent_data['close'].max()
        current_price = recent_data['close'].iloc[-1]
        
        flagpole_growth = (peak_price - start_price) / start_price
        
        # If we had a strong thrust up (> 5%), and are currently resting slightly below the peak
        if flagpole_growth > 0.05 and (current_price < peak_price) and (current_price > start_price):
             print(f"📈 PATTERN DETECTED: Bull Flag forming after {flagpole_growth*100:.1f}% thrust.")
             return True
             
        return False

    def scan_all_patterns(self, df: pd.DataFrame) -> List[str]:
        """
        The master method called by the background worker.
        """
        active_patterns = []
        
        if self.detect_head_and_shoulders(df):
            active_patterns.append('pattern_head_shoulders')
            
        if self.detect_bull_flag(df):
            active_patterns.append('pattern_bull_flag')
            
        return active_patterns
