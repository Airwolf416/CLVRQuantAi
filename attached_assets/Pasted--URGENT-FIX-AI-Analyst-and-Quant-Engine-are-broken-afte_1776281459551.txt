1. Update the parser to handle JSON responses. The AI will now return pure JSON (as specified in the prompts above). The parser should:

```javascript
// Example parsing logic — adapt to your actual code structure
function parseAIResponse(responseText) {
  try {
    // Strip any markdown code fences if present
    const cleaned = responseText.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();
    const parsed = JSON.parse(cleaned);
    
    // Handle both single object and array responses
    const signals = Array.isArray(parsed) ? parsed : [parsed];
    
    return signals.map(signal => {
      if (signal.type === 'signal') {
        return {
          status: 'signal',
          direction: signal.direction,
          asset: signal.asset,
          tradeType: signal.tradeType,
          volRegime: signal.volRegime,
          entry: signal.entry,
          tp1: signal.tp1,
          tp2: signal.tp2,
          tp3: signal.tp3,
          sl: signal.sl,
          rr: signal.rr,
          edge: signal.edge,
          edgeSource: signal.edgeSource,
          killClock: signal.killClock,
          leverage: signal.leverage,
          thesis: signal.thesis,
          invalidation: signal.invalidation,
          postTp1: signal.postTp1,
          macroStatus: signal.macroStatus,
          scores: signal.scores,
          source: signal.source || 'quant'
        };
      } else if (signal.type === 'no_signal') {
        return {
          status: 'no_signal',
          asset: signal.asset,
          reason: signal.reason,
          nextCheck: signal.nextCheck,
          source: signal.source || 'quant'
        };
      } else if (signal.type === 'analysis') {
        return {
          status: 'analysis',
          asset: signal.asset,
          timeframe: signal.timeframe,
          bias: signal.bias,
          support: signal.support,
          resistance: signal.resistance,
          structure: signal.structure,
          flow: signal.flow,
          macro: signal.macro,
          playbook: signal.playbook
        };
      }
      return { status: 'error', message: 'Unknown response type' };
    });
  } catch (e) {
    console.error('Failed to parse AI response:', e);
    console.error('Raw response:', responseText);
    return [{ status: 'error', message: 'AI returned malformed data. Raw: ' + responseText.substring(0, 200) }];
  }
}
```
