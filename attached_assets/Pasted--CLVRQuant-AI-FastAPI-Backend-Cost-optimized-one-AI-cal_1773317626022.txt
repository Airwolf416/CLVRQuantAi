# SHARED SIGNAL CACHE

# AI runs once every 60s server-side. Every user reads the same cached result.

# Cost: 1 AI call per minute total, regardless of how many users are logged in.

class SignalCache:
def **init**(self, ttl_seconds: int = 60):
self.ttl        = ttl_seconds
self.data       = {}        # { asset: signal_dict }
self.market     = {}        # latest market data snapshot
self.generated_at = 0.0     # unix timestamp
self.generating = False     # prevents concurrent AI calls
self.call_count = 0         # track total AI calls made (cost monitoring)

```
def is_stale(self) -> bool:
    return (time.time() - self.generated_at) > self.ttl

def get(self, asset: str) -> dict | None:
    return self.data.get(asset.upper())

def get_all(self) -> dict:
    return {
        "signals":      self.data,
        "market":       self.market,
        "generated_at": self.generated_at,
        "age_seconds":  round(time.time() - self.generated_at, 1),
        "ttl_seconds":  self.ttl,
        "ai_calls_made": self.call_count,
    }

def set_all(self, signals: dict, market: dict):
    self.data         = signals
    self.market       = market
    self.generated_at = time.time()
    self.call_count  += 1
```

async def fetch_market_data() -> dict:
“”“Fetch live prices from Hyperliquid — fast, no AI cost.”””
import httpx
try:
async with httpx.AsyncClient(timeout=8) as client:
r = await client.post(
“https://api.hyperliquid.xyz/info”,
json={“type”: “metaAndAssetCtxs”},
)
data = r.json()
universe = data[0].get(“universe”, [])
ctxs     = data[1] if len(data) > 1 else []
result   = {}
for i, asset in enumerate(universe):
ctx  = ctxs[i] if i < len(ctxs) else {}
name = asset.get(“name”,””)
result[name] = {
“price”:   float(ctx.get(“markPx”, 0)),
“funding”: float(ctx.get(“funding”, 0)),
“change”:  0.0,  # compute from prevDayPx if needed
}
return result
except Exception as e:
return {“error”: str(e)}

async def refresh_cache():
“””
Full refresh: fetch market data + run AI once.
All users read the result — zero per-user AI cost.
“””
if cache.generating:
return  # already running, skip
cache.generating = True
try:
market  = await fetch_market_data()
signals = await run_ai_model(market)
cache.set_all(signals, market)
print(f”[Cache] Refreshed — AI call #{cache.call_count} | {len(signals)} signals cached”)
except Exception as e:
print(f”[Cache] Refresh error: {e}”)
finally:
cache.generating = False

# ─────────────────────────────────────────────────────────────────────────────

# BACKGROUND SCHEDULER — runs inside the event loop, non-blocking

# This replaces your while True / schedule pattern which blocked FastAPI

# ─────────────────────────────────────────────────────────────────────────────

async def background_scheduler():
“”“Runs forever in the background, refreshing cache when stale.”””
await asyncio.sleep(2)          # small startup delay
while True:
if cache.is_stale():
await refresh_cache()
await asyncio.sleep(10)     # check every 10s, refresh if > 60s old

# ─────────────────────────────────────────────────────────────────────────────

# APP LIFESPAN — starts background task cleanly (replaces @app.on_event)

# ─────────────────────────────────────────────────────────────────────────────

@asynccontextmanager
async def lifespan(app: FastAPI):
# Start background cache refresh loop
task = asyncio.create_task(background_scheduler())
print(”[Startup] Background signal scheduler started”)
yield
# Shutdown: cancel the background task cleanly
task.cancel()
try:
await task
except asyncio.CancelledError:
pass
print(”[Shutdown] Scheduler stopped”)

# AUTH — JWT bearer token

# Cheap to verify (pure CPU, no DB hit per request)

# ─────────────────────────────────────────────────────────────────────────────

JWT_SECRET    = os.environ.get(“JWT_SECRET”, “change-this-in-prod”)
JWT_ALGORITHM = “HS256”
security      = HTTPBearer()

def create_token(user_id: str, plan: str = “free”) -> str:
payload = {
“sub”:  user_id,
“plan”: plan,
“iat”:  int(time.time()),
“exp”:  int(time.time()) + 86400 * 7,  # 7 days
}
return jwt.encode(payload, JWT_SECRET, algorithm=JWT_ALGORITHM)

def verify_token(credentials: HTTPAuthorizationCredentials = Depends(security)) -> dict:
try:
payload = jwt.decode(
credentials.credentials,
JWT_SECRET,
algorithms=[JWT_ALGORITHM],
)
return payload
except jwt.ExpiredSignatureError:
raise HTTPException(status_code=401, detail=“Token expired”)
except jwt.InvalidTokenError:
raise HTTPException(status_code=401, detail=“Invalid token”)

def require_pro(user: dict = Depends(verify_token)) -> dict:
if user.get(“plan”) not in (“pro”, “elite”):
raise HTTPException(status_code=403, detail=“Pro plan required”)
return user
