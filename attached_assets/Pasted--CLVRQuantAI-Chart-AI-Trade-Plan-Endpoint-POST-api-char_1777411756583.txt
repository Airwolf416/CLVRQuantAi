CLVRQuantAI — Chart AI Trade Plan Endpoint

POST /api/chartai/plan

- Auth: Elite tier required
- Quota: 5/day per user (resets 00:00 UTC)
- Pulls live snapshot from MarketDataBus singleton
- Calls Claude Sonnet 4.6 with cached system prompt + structured payload + optional chart image
- Returns parsed TradePlan or structured Refusal

class PlanRequest(BaseModel):
ticker: str = Field(…, min_length=1, max_length=12)

class TradePlan(BaseModel):

refusal_code: Optional[str] = None
explanation: Optional[str] = None

class PlanResponse(BaseModel):

@router.post(”/plan”, response_model=PlanResponse)
async def generate_plan(
req: PlanRequest,
user: User = Depends(get_current_user),
bus: MarketDataBus = Depends(get_bus),
):
request_id = uuid4().hex[:12]
started = datetime.now(timezone.utc)

```
# --- Tier check ---
if user.tier != "elite":
    raise HTTPException(403, "Chart AI is an Elite-tier feature.")

# --- Quota check ---
try:
    quota_remaining = await check_and_decrement_quota(
        user.id, "chartai_plan", daily_limit=DAILY_QUOTA
    )
except QuotaExceeded:
    raise HTTPException(429, "Daily Chart AI quota exhausted. Resets at 00:00 UTC.")

# --- Snapshot from shared bus ---
snapshot = await bus.get_chartai_snapshot(req.ticker.upper())
if snapshot is None:
    raise HTTPException(404, f"No live data for {req.ticker}.")

# Defense-in-depth staleness check (model also enforces via REFUSAL.MISSING_DATA)
age_s = (datetime.now(timezone.utc) - snapshot["as_of"]).total_seconds()
asset_class = snapshot["asset"]["class"]
max_age = 60 if asset_class == "perp" else 300
if age_s > max_age:
    return _wrap(
        request_id, started, quota_remaining,
        TradePlan(
            refusal_code="REFUSAL.MISSING_DATA",
            explanation=f"Snapshot {int(age_s)}s old (max {max_age}s for {asset_class}).",
        ),
    )

# --- Build user message ---
payload_json = json.dumps(snapshot["payload"], default=str, indent=2)
user_content = [{
    "type": "text",
    "text": f"Generate a trade plan from this payload:\n\n{payload_json}",
}]
if req.chart_image_b64:
    user_content.insert(0, {
        "type": "image",
        "source": {
            "type": "base64",
            "media_type": "image/png",
            "data": req.chart_image_b64,
        },
    })

# --- Anthropic call (system prompt cached, ~70% cost reduction on hits) ---
try:
    response = client.messages.create(
        model=ANTHROPIC_MODEL,
        max_tokens=MAX_TOKENS,
        system=[{
            "type": "text",
            "text": SYSTEM_PROMPT,
            "cache_control": {"type": "ephemeral"},
        }],
        messages=[{"role": "user", "content": user_content}],
    )
except Exception:
    logger.exception("Anthropic call failed", extra={"request_id": request_id})
    raise HTTPException(502, "Plan generation upstream error.")

# --- Parse response ---
raw_text = "".join(getattr(b, "text", "") for b in response.content)
plan = _parse_plan_json(raw_text, request_id)

# --- Persist for Alert History / analytics ---
await save_plan(
    user_id=user.id,
    request_id=request_id,
    ticker=req.ticker.upper(),
    snapshot=snapshot["payload"],
    plan=plan.model_dump(),
)

logger.info(
    "chartai_plan_generated",
    extra={
        "request_id": request_id,
        "user_id": user.id,
        "ticker": req.ticker.upper(),
        "refusal_code": plan.refusal_code,
        "bias": plan.bias,
        "direction": plan.direction,
        "conviction": plan.conviction,
        "input_tokens": response.usage.input_tokens,
        "cache_read_tokens": getattr(response.usage, "cache_read_input_tokens", 0),
        "output_tokens": response.usage.output_tokens,
    },
)

return _wrap(request_id, started, quota_remaining, plan)
```

def _wrap(request_id: str, started: datetime, quota: int, plan: TradePlan) -> PlanResponse:
latency_ms = int((datetime.now(timezone.utc) - started).total_seconds() * 1000)
return PlanResponse(
request_id=request_id,
plan=plan,
quota_remaining=quota,
latency_ms=latency_ms,
)
