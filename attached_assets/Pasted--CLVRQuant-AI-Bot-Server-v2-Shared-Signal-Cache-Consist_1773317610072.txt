// PRO PASS VALIDATION
// ─────────────────────────────────────────────────────────────────────────────
const PRO_PASSES = new Map(); // token → { activations: [{deviceId, activatedAt, expiresAt}] }

// Seed with your issued passes
[“CLVRPRO7DAY”, “CLVRPRO7DAY2”, “CLVRPRO7DAY3”].forEach(t => {
PRO_PASSES.set(t, { activations: [], maxActivations: Infinity }); // shared = unlimited
});

function validateProPass(token, deviceId) {
const clean = token.trim().toUpperCase().replace(/\s+/g,””);
const pass  = PRO_PASSES.get(clean);

if (!pass) {
// Also accept any CLVR-prefixed token for future passes
if (/^CLVR[A-Z0-9_-]{4,}$/.test(clean)) {
PRO_PASSES.set(clean, { activations: [], maxActivations: Infinity });
return activatePass(clean, deviceId);
}
return { valid: false, error: “Unknown pass token” };
}
return activatePass(clean, deviceId);
}

function activatePass(token, deviceId) {
const pass       = PRO_PASSES.get(token);
const now        = Date.now();
const expiresAt  = now + 7 * 24 * 60 * 60 * 1000;

// Check if this device already has an active activation
const existing = pass.activations.find(a => a.deviceId === deviceId && a.expiresAt > now);
if (existing) {
return { valid: true, expiresAt: existing.expiresAt, daysLeft: Math.ceil((existing.expiresAt - now) / 86400000), existing: true };
}

// New activation for this device
pass.activations.push({ deviceId, activatedAt: now, expiresAt });
return { valid: true, expiresAt, daysLeft: 7, existing: false };
}

async function finnhubQuote(symbol) {
try {
const r = await fetch(`${FINNHUB}/quote?symbol=${encodeURIComponent(symbol)}&token=${FINNHUB_KEY}`);
const d = await r.json();
return d.c ? { price:d.c, change:d.dp } : null;
} catch { return null; }
}

async function fetchNews() {
const headlines = [];
let bullish = 0, total = 0;
try {
// CryptoPanic
if (CRYPTOPANIC_KEY) {
const r = await fetch(`https://cryptopanic.com/api/v1/posts/?auth_token=${CRYPTOPANIC_KEY}&public=true&kind=news&currencies=BTC,ETH,SOL`);
const d = await r.json();
(d?.results||[]).slice(0,8).forEach(p => {
const pos = p.votes?.positive||0, neg = p.votes?.negative||0;
headlines.push({ title:p.title, source:p.source?.title, sentiment: pos>neg?“bullish”:neg>pos?“bearish”:“neutral” });
if (pos>neg) bullish++; total++;
});
}
// Finnhub crypto news
const r2 = await fetch(`${FINNHUB}/news?category=crypto&token=${FINNHUB_KEY}`);
const d2 = await r2.json();
(d2||[]).slice(0,5).forEach(n => headlines.push({ title:n.headline, source:n.source, sentiment:“neutral” }));
} catch {}
const score = total > 0 ? Math.round((bullish/total)*100) : 50;
return { headlines:headlines.slice(0,10), score, label: score>60?“BULLISH”:score<40?“BEARISH”:“MIXED” };
}

// ─────────────────────────────────────────────────────────────────────────────
// MARKET MODE COMPUTATION (deterministic — same inputs = same output always)
// ─────────────────────────────────────────────────────────────────────────────
function computeMarketMode(market) {
const { funding, spy, qqq, vix, gold, eurusd, usdjpy, news } = market;
const vixPx   = vix?.price || 20;
const btcF    = funding?.BTC?.funding || 0;
const usdStrong = (usdjpy?.change||0) > 0.5;

const cryptoBull = Math.round(
(btcF < -0.0005 ? 75 : btcF > 0.001 ? 30 : 50) * 0.5 +
(news.score) * 0.5
);
const equitiesBull = Math.round(
((spy?.change||0) > 1 ? 72 : (spy?.change||0) < -1 ? 30 : 50) * 0.4 +
((qqq?.change||0) > 1 ? 72 : (qqq?.change||0) < -1 ? 30 : 50) * 0.4 +
(vixPx < 16 ? 75 : vixPx < 22 ? 50 : vixPx < 30 ? 35 : 20) * 0.2
);
const metalsBull = Math.round((gold?.change||0) > 1 ? 65 : (gold?.change||0) < -1 ? 35 : 50);
const forexBull  = Math.round(usdStrong ? 35 : 65);

const master = Math.round(cryptoBull*0.35 + equitiesBull*0.35 + metalsBull*0.15 + forexBull*0.15);

return {
score:     master,
regime:    master >= 60 ? “RISK-ON” : master <= 40 ? “RISK-OFF” : “NEUTRAL”,
crypto:    { bull:cryptoBull,   bear:100-cryptoBull,   regime:cryptoBull>=60?“BULLISH”:cryptoBull<=40?“BEARISH”:“NEUTRAL” },
equities:  { bull:equitiesBull, bear:100-equitiesBull, regime:equitiesBull>=60?“BULLISH”:equitiesBull<=40?“BEARISH”:“NEUTRAL” },
metals:    { bull:metalsBull,   bear:100-metalsBull,   regime:metalsBull>=60?“BULLISH”:metalsBull<=40?“BEARISH”:“NEUTRAL” },
forex:     { bull:forexBull,    bear:100-forexBull,    regime:forexBull>=60?“BULLISH”:forexBull<=40?“BEARISH”:“NEUTRAL” },
};
}

? account.positions.map(p => `  • ${p.side} ${Math.abs(p.size)} ${p.asset} @ $${p.entryPx?.toFixed(2)} | uPnL: ${p.unrealizedPnl>=0?"+":""}$${p.unrealizedPnl?.toFixed(2)}`).join(”\n”)
: “  None”;

// DETERMINISTIC system prompt — instructs Claude to be consistent

Base direction purely on: funding rate sign, news sentiment score, market regime score, and momentum.

Schema:

“confidence”: 0-100,

“takeProfit”: number,

“keyRisks”: [“risk1”,“risk2”],
“timeframe”: “scalp (mins)”|“intraday (hours)”|“swing (days)”
}`;

const user = `Generate signal: ${asset}/USD perpetual

CLVR MARKET MODE: ${marketMode.regime} (${marketMode.score}% Risk-On)
Crypto Regime:    ${marketMode.crypto.regime} (${marketMode.crypto.bull}% bull)

function parseAccount(raw) {
if (!raw) return null;
const margin = raw.marginSummary || {};
const positions = (raw.assetPositions||[]).map(ap => {
const pos = ap.position;
if (!pos || parseFloat(pos.szi)===0) return null;
const size = parseFloat(pos.szi);
const mu   = parseFloat(pos.marginUsed||0);
const upnl = parseFloat(pos.unrealizedPnl||0);
return { asset:pos.coin, size, side:size>0?“LONG”:“SHORT”, entryPx:parseFloat(pos.entryPx||0), unrealizedPnl:upnl, leverage:parseFloat(pos.leverage?.value||pos.leverage||1), liqPx:parseFloat(pos.liquidationPx||0), marginUsed:mu, roe:mu>0?(upnl/mu)*100:0 };
}).filter(Boolean);
return { accountValue:parseFloat(margin.accountValue||0), totalMarginUsed:parseFloat(margin.totalMarginUsed||0), totalUnrealizedPnl:parseFloat(margin.totalUnrealizedPnl||0), withdrawable:parseFloat(raw.withdrawable||0), positions };
}

// ─────────────────────────────────────────────────────────────────────────────
// ORDER SIGNING
// ─────────────────────────────────────────────────────────────────────────────
async function submitHLOrder({ assetIndex, isBuy, size, price=0, leverage }) {
if (!PRIVATE_KEY) throw new Error(“HL_PRIVATE_KEY not set”);
const wallet    = new ethers.Wallet(PRIVATE_KEY);
const timestamp = Date.now();
const order     = { a:assetIndex, b:isBuy, p:price===0?“0”:price.toString(), s:size.toString(), r:false, t:price===0?{market:{tpsl:””,limitPx:“0”}}:{limit:{tif:“Gtc”}} };
const action    = { type:“order”, orders:[order], grouping:“na” };
const actionBytes = ethers.utils.toUtf8Bytes(JSON.stringify(action));
const actionHash  = ethers.utils.keccak256(actionBytes);
const nonceBytes  = ethers.utils.arrayify(ethers.utils.hexZeroPad(ethers.BigNumber.from(timestamp).toHexString(),8));
const vaultBytes  = ethers.utils.arrayify(“0x0000000000000000000000000000000000000000”);
const msgHash     = ethers.utils.solidityKeccak256([“bytes32”,“bytes8”,“address”],[actionHash,nonceBytes,vaultBytes]);
const sig         = await wallet.signMessage(ethers.utils.arrayify(msgHash));
const { r, s, v } = ethers.utils.splitSignature(sig);
const payload = { action, nonce:timestamp, signature:{r,s,v}, vaultAddress:null };
const res = await fetch(HL_EXCHANGE, { method:“POST”, headers:{“Content-Type”:“application/json”}, body:JSON.stringify(payload) });
const result = await res.json();
if (result.status !== “ok”) throw new Error(result.response||JSON.stringify(result));
return result;
}

function startBot(config, risk) {
if (config) botState.config = { …botState.config, …config };
if (risk)   botState.risk   = { …botState.risk,   …risk   };
if (botState.risk.circuitBreakerActive) return { error:“Reset circuit breaker first” };

runScanCycle();
botState.scanInterval = setInterval(runScanCycle, botState.config.scanIntervalSeconds*1000);
return { success:true };
}

function stopBot() {
clearInterval(botState.scanInterval); botState.scanInterval = null;
botState.status = “idle”;
log(“🛑 Bot stopped”,“warn”);
}

res.json(validateProPass(token, deviceId));
});

// Add new pass token (admin)
app.post(”/pro/add-token”, (req,res) => {
const { token, adminKey } = req.body;
if (adminKey !== process.env.ADMIN_KEY) return res.status(403).json({ error:“Forbidden” });
const clean = token.trim().toUpperCase();
PRO_PASSES.set(clean, { activations:[], maxActivations:Infinity });
res.json({ success:true, token:clean });
});
