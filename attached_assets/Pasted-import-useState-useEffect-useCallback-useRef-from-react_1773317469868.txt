async function hlPost(body) {
try {
const r = await fetch(HL_API, { method:“POST”, headers:{“Content-Type”:“application/json”}, body:JSON.stringify(body) });
return await r.json();
} catch { return null; }
}

async function fetchHLMarket() {
const [mids, meta] = await Promise.all([
hlPost({ type:“allMids” }),
hlPost({ type:“metaAndAssetCtxs” }),
]);
const universe = meta?.[0]?.universe || [];
const ctxs     = meta?.[1] || [];
const funding  = {};
universe.forEach((a,i) => {
funding[a.name] = {
funding:      parseFloat(ctxs[i]?.funding      || 0),
openInterest: parseFloat(ctxs[i]?.openInterest || 0),
markPx:       parseFloat(ctxs[i]?.markPx       || 0),
index: i,
};
});
return { prices: mids || {}, funding };
}

async function fetchHLAccount(evmAddress) {
try {
const [state, orders] = await Promise.all([
hlPost({ type:“clearinghouseState”, user:evmAddress }),
hlPost({ type:“openOrders”,         user:evmAddress }),
]);
const margin = state?.marginSummary || {};
const positions = (state?.assetPositions || []).map(ap => {
const pos = ap.position;
if (!pos || parseFloat(pos.szi) === 0) return null;
const size = parseFloat(pos.szi);
const mu   = parseFloat(pos.marginUsed   || 0);
const upnl = parseFloat(pos.unrealizedPnl || 0);
return {
asset: pos.coin, size, side: size > 0 ? “LONG” : “SHORT”,
entryPx: parseFloat(pos.entryPx || 0), unrealizedPnl: upnl,
leverage: parseFloat(pos.leverage?.value || pos.leverage || 1),
liqPx: parseFloat(pos.liquidationPx || 0), marginUsed: mu,
roe: mu > 0 ? (upnl / mu) * 100 : 0,
};
}).filter(Boolean);
return {
accountValue:        parseFloat(margin.accountValue       || 0),
totalMarginUsed:     parseFloat(margin.totalMarginUsed    || 0),
totalUnrealizedPnl:  parseFloat(margin.totalUnrealizedPnl || 0),
withdrawable:        parseFloat(state?.withdrawable       || 0),
positions, openOrders: orders || [],
};
} catch { return null; }
}

// ─────────────────────────────────────────────────────────────────────────────
// MARKET MODE ENGINE — calculates Risk-On/Off regime from all data
// ─────────────────────────────────────────────────────────────────────────────
function computeMarketMode({ macro, hlMarket, social }) {
let score = 50; // start neutral
const signals = {};

// VIX: below 18 = calm (risk-on), above 25 = fear (risk-off)
if (macro?.vix?.price) {
const v = macro.vix.price;
signals.vix = v < 18 ? “LOW”  : v < 25 ? “MODERATE” : “HIGH”;
score += v < 18 ? +12 : v < 25 ? 0 : -18;
}

// SPY momentum
if (macro?.spx?.change) {
const c = macro.spx.change;
signals.nasdaq = c > 1 ? “BULLISH” : c > -1 ? “NEUTRAL” : “BEARISH”;
score += c > 1 ? +10 : c > -1 ? 0 : -10;
}

// Oil — rising oil can signal risk-off / inflation
if (macro?.oil?.change) {
const c = macro.oil.change;
signals.oil = c > 2 ? “BEARISH” : c > -1 ? “NEUTRAL” : “BULLISH”;
score += c > 2 ? -6 : c > -1 ? 0 : +4;
}

// Nasdaq (QQQ) — direct AI/tech proxy
if (macro?.ndx?.change) {
const c = macro.ndx.change;
signals.aiStocks = c > 1.5 ? “BULLISH” : c > -1 ? “NEUTRAL” : “BEARISH”;
score += c > 1.5 ? +8 : c > -1 ? 0 : -8;
}

// BTC funding — positive = overheated (slight bearish), negative = underinvested (bullish)
const btcFunding = hlMarket?.funding?.BTC?.funding || 0;
signals.btcFunding = btcFunding > 0.001 ? “OVERHEATED” : btcFunding < -0.0005 ? “OVERSOLD” : “NEUTRAL”;
score += btcFunding > 0.001 ? -5 : btcFunding < -0.0005 ? +5 : 0;

// Social sentiment
if (social?.score) {
score += (social.score - 50) * 0.15;
signals.social = social.label;
}

// Clamp
score = Math.max(0, Math.min(100, score));

const riskOn  = score;
const regime  = score >= 60 ? “RISK-ON” : score <= 40 ? “RISK-OFF” : “NEUTRAL”;
const btcBull = Math.min(95, Math.max(5, score + (btcFunding < 0 ? 8 : -5)));

return { score: Math.round(riskOn), regime, signals, btcBull: Math.round(btcBull), btcBear: 100 - Math.round(btcBull) };
}

// ─────────────────────────────────────────────────────────────────────────────
// RISK ENGINE
// ─────────────────────────────────────────────────────────────────────────────
function evaluateRisk(signal, account, riskConfig, marketMode) {
const violations = [];

if (signal.confidence < riskConfig.minConfidence)
violations.push(`Confidence ${signal.confidence}% < min ${riskConfig.minConfidence}%`);

if (account && account.totalUnrealizedPnl < -riskConfig.dailyLossLimit)
violations.push(`Drawdown $${Math.abs(account.totalUnrealizedPnl).toFixed(0)} > limit $${riskConfig.dailyLossLimit}`);

if (account && account.positions.length >= riskConfig.maxPositions)
violations.push(`Max ${riskConfig.maxPositions} positions reached`);

if (account) {
const dup = account.positions.find(p => p.asset === signal.asset && p.side === signal.direction);
if (dup) violations.push(`Already ${signal.direction} ${signal.asset}`);
}

if (signal.fundingRate) {
const adverse = (signal.direction === “LONG” && signal.fundingRate > 0) ||
(signal.direction === “SHORT” && signal.fundingRate < 0);
if (adverse && Math.abs(signal.fundingRate) > riskConfig.maxFundingRate)
violations.push(`Adverse funding ${(signal.fundingRate*100).toFixed(3)}%`);
}

if (signal.signal === “NEUTRAL”)
violations.push(“Neutral signal — no trade”);

if (riskConfig.pauseOnRiskOff && marketMode?.regime === “RISK-OFF” && signal.direction === “LONG”)
violations.push(“RISK-OFF regime — LONG entries paused”);

return { passed: violations.length === 0, violations };
}
