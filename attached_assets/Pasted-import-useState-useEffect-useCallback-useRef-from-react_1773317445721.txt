async function buildHLOrderPayload({ asset, isBuy, size, price, leverage, reduceOnly = false }) {
// Hyperliquid asset index lookup
const metaRes = await fetch(HL_API, {
method: “POST”,
headers: { “Content-Type”: “application/json” },
body: JSON.stringify({ type: “meta” }),
});
const meta = await metaRes.json();
const assetIndex = meta.universe.findIndex(a => a.name === asset);
if (assetIndex === -1) throw new Error(`Asset ${asset} not found on HL`);

const timestamp = Date.now();
const order = {
a: assetIndex,           // asset index
b: isBuy,                // is buy
p: price.toString(),     // price (use “0” for market)
s: size.toString(),      // size
r: reduceOnly,           // reduce only
t: price === 0           // order type: market or limit
? { market: {} }
: { limit: { tif: “Gtc” } },
};

return { order, timestamp, assetIndex };
}

function evaluateRisk(signal, hlAccount, riskConfig) {
const violations = [];

// 1. Confidence threshold
if (signal.confidence < riskConfig.minConfidence) {
violations.push(`Confidence ${signal.confidence}% below minimum ${riskConfig.minConfidence}%`);
}

// 2. Daily loss limit
if (hlAccount && Math.abs(hlAccount.totalUnrealizedPnl) > riskConfig.dailyLossLimit
&& hlAccount.totalUnrealizedPnl < 0) {
violations.push(`Daily loss $${Math.abs(hlAccount.totalUnrealizedPnl).toFixed(2)} exceeds limit $${riskConfig.dailyLossLimit}`);
}

// 3. Max concurrent positions
if (hlAccount && hlAccount.positions.length >= riskConfig.maxPositions) {
violations.push(`Already at max ${riskConfig.maxPositions} open positions`);
}

// 4. Duplicate position check
if (hlAccount) {
const existing = hlAccount.positions.find(p => p.asset === signal.asset);
if (existing && existing.side === signal.direction) {
violations.push(`Already ${signal.direction} ${signal.asset} — duplicate signal blocked`);
}
}

// 5. Funding rate filter
if (signal.fundingRate && Math.abs(signal.fundingRate) > riskConfig.maxFundingRate) {
const fundingAdverse = (signal.direction === “LONG” && signal.fundingRate > 0) ||
(signal.direction === “SHORT” && signal.fundingRate < 0);
if (fundingAdverse) {
violations.push(`Funding rate ${(signal.fundingRate*100).toFixed(4)}% adverse — exceeds ${(riskConfig.maxFundingRate*100).toFixed(3)}% limit`);
}
}

// 6. Leverage cap
if (signal.leverage > riskConfig.maxLeverage) {
violations.push(`Leverage ${signal.leverage}x exceeds cap ${riskConfig.maxLeverage}x`);
}

// 7. Neutral signal block
if (signal.signal === “NEUTRAL”) {
violations.push(“AI signal is NEUTRAL — no trade taken”);
}

return { passed: violations.length === 0, violations };
}
