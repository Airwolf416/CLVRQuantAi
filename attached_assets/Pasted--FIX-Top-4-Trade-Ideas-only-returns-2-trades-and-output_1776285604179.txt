## BUG 2: Output looks like ugly raw markdown — needs to be parsed and styled

### Root Cause

The AI response (whether markdown or JSON) is being dumped directly into the UI as raw text. It needs to be parsed into structured UI components.

### Fix

Find the component that renders the Top Trade Ideas output. It’s currently just displaying the raw AI response text. Replace it with a component that parses the JSON and renders styled cards.

Here is the rendering logic to implement. Adapt it to match your existing design system (dark theme, cyan/green/gold accents that I can see in the screenshots):

```jsx
// TradeIdeasDisplay component — renders parsed JSON trade ideas
// Adapt class names and styles to match your existing CLVRQuant design system

function TradeIdeasDisplay({ rawResponse }) {
  const [data, setData] = useState(null);
  const [error, setError] = useState(null);

  useEffect(() => {
    try {
      const cleaned = rawResponse.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();
      setData(JSON.parse(cleaned));
    } catch (e) {
      setError('Failed to parse trade ideas');
    }
  }, [rawResponse]);

  if (error) return <div className="error-banner">{error}</div>;
  if (!data) return <div className="loading">Analyzing markets...</div>;

  return (
    <div className="trade-ideas-container">
      {/* Header Bar */}
      <div className="ideas-header">
        <h2>TOP 4 TRADE IDEAS</h2>
        <div className="header-meta">
          <span className="timestamp">{new Date(data.generated).toLocaleString()}</span>
          <span className={`regime-badge ${data.regime.score > 60 ? 'risk-on' : 'risk-off'}`}>
            {data.regime.label} {data.regime.score}/100
          </span>
          <span className={`vol-badge vol-${data.volRegime.toLowerCase()}`}>
            VOL: {data.volRegime}
          </span>
        </div>
      </div>

      {/* Macro Status Bar */}
      <div className="macro-bar">
        <span className={data.macroStatus.clear ? 'macro-clear' : 'macro-warn'}>
          {data.macroStatus.clear ? '✅' : '⚠️'} MACRO: {data.macroStatus.clear ? 'CLEAR' : 'CAUTION'}
        </span>
        {data.macroStatus.nextEvent && (
          <span className="next-event">Next: {data.macroStatus.nextEvent}</span>
        )}
      </div>

      {/* Trade Cards */}
      <div className="trade-cards-grid">
        {data.trades.map((trade, i) => (
          <TradeCard key={i} trade={trade} />
        ))}
      </div>
    </div>
  );
}

function TradeCard({ trade }) {
  const isLong = trade.direction === 'LONG';

  return (
    <div className={`trade-card ${isLong ? 'card-long' : 'card-short'}`}>
      {/* Card Header */}
      <div className="card-header">
        <div className="card-rank">#{trade.rank}</div>
        <div className="card-asset">{trade.asset}</div>
        <div className={`card-direction ${isLong ? 'dir-long' : 'dir-short'}`}>
          {isLong ? '🟢' : '🔴'} {trade.direction}
        </div>
        <div className="card-type">{trade.tradeType}</div>
      </div>

      {/* Conviction Bar */}
      <div className="conviction-bar">
        <div className="conviction-fill" style={{ width: `${trade.conviction}%` }}>
          {trade.conviction}%
        </div>
      </div>

      {/* Levels Grid */}
      <div className="levels-grid">
        <div className="level-row entry-row">
          <span className="level-label">ENTRY</span>
          <span className="level-price">${trade.entry}</span>
        </div>
        <div className="level-row tp1-row">
          <span className="level-label">TP1 (50%)</span>
          <span className="level-price">${trade.tp1.price}</span>
          <span className="level-rr">{trade.tp1.rr}</span>
        </div>
        <div className="level-row tp2-row">
          <span className="level-label">TP2 (30%)</span>
          <span className="level-price">${trade.tp2.price}</span>
          <span className="level-rr">{trade.tp2.rr}</span>
        </div>
        <div className="level-row tp3-row">
          <span className="level-label">TP3 (20%)</span>
          <span className="level-price">${trade.tp3.price}</span>
          <span className="level-tag">TRAIL</span>
        </div>
        <div className="level-row sl-row">
          <span className="level-label">STOP</span>
          <span className="level-price sl-price">${trade.sl}</span>
        </div>
      </div>

      {/* Meta Row */}
      <div className="meta-row">
        <span className="meta-item">⚡ {trade.leverage}</span>
        <span className="meta-item">⏱ {trade.killClock}</span>
        <span className="meta-item">📊 {trade.edge} ({trade.edgeSource})</span>
      </div>

      {/* Thesis */}
      <div className="thesis-box">
        <p className="thesis-text">{trade.thesis}</p>
      </div>

      {/* Invalidation */}
      <div className="invalidation-box">
        <span className="inv-label">INVALIDATION:</span> {trade.invalidation}
      </div>

      {/* Flags */}
      {trade.flags && trade.flags.length > 0 && (
        <div className="flags-box">
          {trade.flags.map((flag, i) => (
            <span key={i} className="flag-item">⚠️ {flag}</span>
          ))}
        </div>
      )}

      {/* Factor Scores Mini Bar */}
      {trade.scores && (
        <div className="scores-row">
          {Object.entries(trade.scores).map(([key, val]) => (
            <div key={key} className="score-item">
              <span className="score-label">{key.substring(0, 4).toUpperCase()}</span>
              <div className="score-bar">
                <div
                  className={`score-fill ${val >= 70 ? 'score-high' : val >= 50 ? 'score-mid' : 'score-low'}`}
                  style={{ width: `${val}%` }}
                />
              </div>
              <span className="score-val">{val}</span>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```
