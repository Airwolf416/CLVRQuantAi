async function callAIWithRetry(system, userMessage, maxTokens, retries = 3) {
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      const response = await callClaudeAPI(system, userMessage, maxTokens);
      if (response && response.text) return response;
    } catch (err) {
      console.error(`[AI] Attempt ${attempt}/${retries} failed:`, err.message);
      if (attempt < retries) {
        const delay = Math.pow(2, attempt) * 1000; // 2s, 4s, 8s
        await new Promise(r => setTimeout(r, delay));
      }
    }
  }
  return null; // All retries failed
}

app.get("/api/journal/summary", async (req, res) => {
  const user = await getUser(req.session?.userId);
  if (!user) return res.status(401).json({ error: "Not authenticated" });

  const period = req.query.period || 'weekly'; // daily, weekly, monthly
  const now = new Date();
  let startDate;

  if (period === 'daily') startDate = new Date(now.setHours(0,0,0,0));
  else if (period === 'weekly') startDate = new Date(now.setDate(now.getDate() - 7));
  else startDate = new Date(now.setMonth(now.getMonth() - 1));

  const trades = await db.select().from(journal_trades)
    .where(and(
      eq(journal_trades.user_id, user.id),
      sql`${journal_trades.trade_date} >= ${startDate}`
    ));

  const wins = trades.filter(t => t.pnl_pct > 0);
  const losses = trades.filter(t => t.pnl_pct <= 0);
  const totalPnl = trades.reduce((sum, t) => sum + (parseFloat(t.pnl_pct) || 0), 0);
  const best = trades.reduce((b, t) => (!b || parseFloat(t.pnl_pct) > parseFloat(b.pnl_pct)) ? t : b, null);
  const worst = trades.reduce((w, t) => (!w || parseFloat(t.pnl_pct) < parseFloat(w.pnl_pct)) ? t : w, null);

  res.json({
    period,
    trades: trades.length,
    winRate: trades.length > 0 ? Math.round((wins.length / trades.length) * 100) : 0,
    totalPnl: Math.round(totalPnl * 100) / 100,
    avgPnl: trades.length > 0 ? Math.round((totalPnl / trades.length) * 100) / 100 : 0,
    best: best ? { asset: best.asset, pnl: best.pnl_pct } : null,
    worst: worst ? { asset: worst.asset, pnl: worst.pnl_pct } : null,
    wins: wins.length,
    losses: losses.length,
  });
});
