// ─── REGIME ALIGNMENT GATE ────────────────────────────────────────────────────
// Verifies AI recommendation aligns with deterministic regime conditions.
// 6 hard checks → ALIGNED publishes full, PARTIAL halves leverage, MISALIGNED blocks.
function computeRegimeGate(
  ind: any,
  confluence: any,
  bayesian: any,
  macroKillSwitch: any,
  aiSignal: string  // "STRONG_LONG" | "LONG" | "NEUTRAL" | "SHORT" | "STRONG_SHORT"
) {
  const sig = (aiSignal || "NEUTRAL").toUpperCase();
  const isLong  = sig === "LONG"  || sig === "STRONG_LONG";
  const isShort = sig === "SHORT" || sig === "STRONG_SHORT";
  const dir = isLong ? "LONG" : isShort ? "SHORT" : "NEUTRAL";

  // Hard block: macro kill switch overrides everything
  if (macroKillSwitch?.active) {
    return {
      verdict: "BLOCK",
      action: "BLOCK",
      direction: dir,
      score: 0,
      reason: `Macro kill switch active: ${macroKillSwitch.reason || "HIGH-impact event within 4h"}`,
      checks: [{ name: "Macro window", pass: false, detail: macroKillSwitch.reason || "HIGH event imminent" }],
      adjustments: { leverageMultiplier: 0, convictionCap: "D" },
    };
  }

  // AI itself said NEUTRAL → pass through, gate is informational only
  if (dir === "NEUTRAL") {
    return {
      verdict: "PASS_THROUGH",
      action: "PUBLISH",
      direction: "NEUTRAL",
      score: 50,
      reason: "AI returned NEUTRAL — no trade to gate",
      checks: [],
      adjustments: { leverageMultiplier: 1, convictionCap: bayesian?.tier || "C" },
    };
  }

  const checks: { name: string; pass: boolean; detail: string }[] = [];

  // 1. Trend alignment (ind.trend must agree with AI direction)
  const trend = ind.trend || "";
  const trendOk = isLong ? /UPTREND/.test(trend) : /DOWNTREND/.test(trend);
  checks.push({
    name: "Trend",
    pass: trendOk,
    detail: `${trend || "n/a"} ${trendOk ? "✓ aligns" : "✗ fights"} ${dir}`,
  });

  // 2. Multi-TF confluence (4h + 1d agreement)
  const confDir = confluence?.direction || "MIXED";
  const confOk = isLong
    ? confDir === "BULLISH" || confDir === "LEANING_BULL"
    : confDir === "BEARISH" || confDir === "LEANING_BEAR";
  checks.push({
    name: "Multi-TF confluence",
    pass: confOk,
    detail: `${confDir} (${confluence?.strength || "n/a"})`,
  });

  // 3. EMA stack (price > EMA20 > EMA50 > EMA200, or inverse)
  const stackOk = isLong
    ? ind.currentPrice > ind.ema20 && ind.ema20 > ind.ema50 && ind.ema50 > ind.ema200
    : ind.currentPrice < ind.ema20 && ind.ema20 < ind.ema50 && ind.ema50 < ind.ema200;
  checks.push({
    name: "EMA stack",
    pass: stackOk,
    detail: stackOk ? "Stack confirms direction" : "Stack broken or mixed",
  });

  // 4. Momentum agreement
  const mom = ind.momentumScore ?? 50;
  const momOk = isLong ? mom >= 55 : mom <= 45;
  checks.push({
    name: "Momentum",
    pass: momOk,
    detail: `${mom}/100 ${momOk ? "✓" : "✗ neutral or opposing"}`,
  });

  // 5. RSI not exhausted against the direction
  const rsi = ind.rsi ?? 50;
  const rsiOk = isLong ? rsi < 75 : rsi > 25;
  checks.push({
    name: "RSI exhaustion",
    pass: rsiOk,
    detail: `RSI ${rsi} ${rsiOk ? "ok" : "exhausted — reversal risk"}`,
  });

  // 6. Volatility regime sanity (ATR% in usable band)
  const atrPct = parseFloat(ind.atrPct) || 0;
  const volOk = atrPct >= 0.3 && atrPct <= 12;
  checks.push({
    name: "Volatility regime",
    pass: volOk,
    detail: `ATR ${atrPct.toFixed(2)}% ${volOk ? "tradeable" : atrPct < 0.3 ? "too quiet — SL hits in noise" : "too wild — slippage risk"}`,
  });

  const passCount = checks.filter(c => c.pass).length;
  const total = checks.length;
  const score = Math.round((passCount / total) * 100);

  let verdict: string, action: string, leverageMultiplier: number, convictionCap: string;
  if (passCount === total) {
    verdict = "ALIGNED";
    action = "PUBLISH";
    leverageMultiplier = 1;
    convictionCap = bayesian?.tier || "B";
  } else if (passCount >= 4) {
    verdict = "PARTIAL";
    action = "DOWNGRADE";
    leverageMultiplier = 0.5;
    convictionCap = "C";
  } else {
    verdict = "MISALIGNED";
    action = "BLOCK";
    leverageMultiplier = 0;
    convictionCap = "D";
  }

  return {
    verdict,
    action,
    direction: dir,
    score,
    reason: `${passCount}/${total} regime checks passed`,
    checks,
    adjustments: { leverageMultiplier, convictionCap },
  };
}
