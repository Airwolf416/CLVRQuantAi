// Handles the 60-second dropdown and the "Silent Unlock" so the bell rings on iOS/Chrome
// ============================================================================
import React, { useState, useEffect, useRef } from 'react';

const MarketClock = () => {
  const [countdown, setCountdown] = useState(null);
  const bellPlayed = useRef(false);
  const bellAudio = useRef(typeof Audio !== "undefined" ? new Audio('/bell.mp3') : null);

  useEffect(() => {
    // SILENT UNLOCK: Forces browser to authorize audio on the first tap/click
    const unlockAudio = () => {
      if (bellAudio.current) {
        bellAudio.current.volume = 0; 
        bellAudio.current.play().then(() => {
          bellAudio.current.pause();
          bellAudio.current.currentTime = 0; 
          bellAudio.current.volume = 1.0;    
        }).catch(e => console.log("Audio unlock pending..."));
      }
      document.removeEventListener('click', unlockAudio);
      document.removeEventListener('touchstart', unlockAudio);
    };

    document.addEventListener('click', unlockAudio);
    document.addEventListener('touchstart', unlockAudio);

    const checkTime = () => {
      const now = new Date(new Date().toLocaleString("en-US", {timeZone: "America/New_York"}));
      const day = now.getDay();
      if (day === 0 || day === 6) return;

      const hours = now.getHours();
      const minutes = now.getMinutes();
      const seconds = now.getSeconds();
      const currentTimeInSeconds = (hours * 3600) + (minutes * 60) + seconds;

      const openTime = (9 * 3600) + (30 * 60); // 9:30:00 AM ET
      const closeTime = 16 * 3600;             // 4:00:00 PM ET

      let targetTime = null;
      let eventType = '';

      if (currentTimeInSeconds < openTime) {
        targetTime = openTime;
        eventType = 'OPEN';
      } else if (currentTimeInSeconds < closeTime) {
        targetTime = closeTime;
        eventType = 'CLOSE';
      } else {
        bellPlayed.current = false; 
        return;
      }

      const diff = targetTime - currentTimeInSeconds;

      if (diff <= 60 && diff > 0) {
        setCountdown({ seconds: diff, type: eventType });
        bellPlayed.current = false;
      } else if (diff === 0 && !bellPlayed.current) {
        setCountdown(null);
        if (bellAudio.current) {
          bellAudio.current.play().catch(e => console.log("Autoplay blocked.", e));
        }
        bellPlayed.current = true;
      } else {
        setCountdown(null);
      }
    };

    const interval = setInterval(checkTime, 1000);
    return () => {
      clearInterval(interval);
      document.removeEventListener('click', unlockAudio);
      document.removeEventListener('touchstart', unlockAudio);
    };
  }, []);

  if (!countdown) return null;

  return (
    <div style={{
      position: 'fixed', top: 0, left: 0, width: '100%', padding: '12px',
      background: 'linear-gradient(135deg, #d4af37, #f5d062)', color: '#060a13',
      fontWeight: '900', textAlign: 'center', fontFamily: "'IBM Plex Mono', monospace",
      boxShadow: '0 4px 20px rgba(212,175,55,0.4)', zIndex: 9999,
      animation: 'slideDown 0.4s ease-out'
    }}>
      🔔 WALL STREET WILL {countdown.type} IN {countdown.seconds} SECONDS 🔔
      <style>{`@keyframes slideDown { from { transform: translateY(-100%); } to { transform: translateY(0); } }`}</style>
    </div>
  );
};

export default MarketClock;

// BACKEND FILE 5: server/routes/webhook.js
// Zero-touch database upgrades via Stripe Webhooks
// ============================================================================
const express = require('express');
const router = express.Router();
const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);
const db = require('../db'); 

// CRITICAL: Must use express.raw for Stripe signatures
router.post('/stripe', express.raw({ type: 'application/json' }), async (req, res) => {
  const signature = req.headers['stripe-signature'];
  let event;

  try {
    event = stripe.webhooks.constructEvent(req.body, signature, process.env.STRIPE_WEBHOOK_SECRET);
  } catch (err) {
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }

  if (event.type === 'checkout.session.completed') {
    const session = event.data.object;
    const userId = session.client_reference_id; 
    let newTier = session.metadata.plan_type === 'elite' ? 'elite' : 'pro';

    await db.query(
      `UPDATE users SET subscription_status = $1, stripe_customer_id = $2, updated_at = CURRENT_TIMESTAMP WHERE id = $3`,
      [newTier, session.customer, userId]
    );
  }

  if (event.type === 'customer.subscription.deleted') {
    const subscription = event.data.object;
    await db.query(`UPDATE users SET subscription_status = 'free' WHERE stripe_customer_id = $1`, [subscription.customer]);
  }

  res.json({ received: true });
});

module.exports = router;
