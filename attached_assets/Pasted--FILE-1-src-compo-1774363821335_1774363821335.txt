import React, { useState, useEffect, useRef } from 'react';

const MarketClock = () => {
  const [countdown, setCountdown] = useState(null);
  const bellPlayed = useRef(false);

  useEffect(() => {
    const checkTime = () => {
      // Force exact Wall Street Time (Eastern Time)
      const now = new Date(new Date().toLocaleString("en-US", {timeZone: "America/New_York"}));
      const day = now.getDay();
      
      // Skip weekends
      if (day === 0 || day === 6) return;

      const hours = now.getHours();
      const minutes = now.getMinutes();
      const seconds = now.getSeconds();
      const currentTimeInSeconds = (hours * 3600) + (minutes * 60) + seconds;

      // --- STANDARD LIVE MODE ---
      const openTime = (9 * 3600) + (30 * 60); // 9:30:00 AM
      const closeTime = 16 * 3600;             // 4:00:00 PM

      // --- TEST MODE OVERRIDE (Uncomment the 2 lines below to test the bell right now) ---
      // const openTime = currentTimeInSeconds + 65; 
      // const closeTime = currentTimeInSeconds + 120;

      let targetTime = null;
      let eventType = '';

      if (currentTimeInSeconds < openTime) {
        targetTime = openTime;
        eventType = 'OPEN';
      } else if (currentTimeInSeconds < closeTime) {
        targetTime = closeTime;
        eventType = 'CLOSE';
      } else {
        bellPlayed.current = false; 
        return;
      }

      const diff = targetTime - currentTimeInSeconds;

      if (diff <= 60 && diff > 0) {
        setCountdown({ seconds: diff, type: eventType });
        bellPlayed.current = false;
      } else if (diff === 0 && !bellPlayed.current) {
        setCountdown(null);
        
        // Make sure bell.mp3 is in your public folder!
        const audio = new Audio('/bell.mp3'); 
        audio.volume = 1.0;
        audio.play().catch(e => console.log("Autoplay blocked: User must interact with page first.", e));
        
        bellPlayed.current = true;
      } else {
        setCountdown(null);
      }
    };

    const interval = setInterval(checkTime, 1000);
    return () => clearInterval(interval);
  }, []);

  if (!countdown) return null;

  return (
    <div style={{
      position: 'fixed', top: 0, left: 0, width: '100%', padding: '12px',
      background: 'linear-gradient(135deg, #d4af37, #f5d062)', color: '#060a13',
      fontWeight: '900', textAlign: 'center', fontFamily: "'IBM Plex Mono', monospace",
      boxShadow: '0 4px 20px rgba(212,175,55,0.4)', zIndex: 9999,
      animation: 'slideDown 0.4s ease-out'
    }}>
      🔔 WALL STREET WILL {countdown.type} IN {countdown.seconds} SECONDS 🔔
      <style>{`
        @keyframes slideDown {
          from { transform: translateY(-100%); }
          to { transform: translateY(0); }
        }
      `}</style>
    </div>
  );
};

export default MarketClock;

import React, { useState, useEffect, useCallback } from 'react';

const SquawkBox = ({ latestSignal }) => {
  const [isMuted, setIsMuted] = useState(true);
  const [voice, setVoice] = useState(null);

  useEffect(() => {
    const loadVoices = () => {
      const availableVoices = window.speechSynthesis.getVoices();
      const preferredVoice = availableVoices.find(
        (v) => v.lang === 'en-US' && (v.name.includes('Google') || v.name.includes('Enhanced'))
      );
      setVoice(preferredVoice || availableVoices[0]);
    };

    loadVoices();
    window.speechSynthesis.onvoiceschanged = loadVoices;
  }, []);

  const triggerSquawk = useCallback((text) => {
    if (isMuted || !window.speechSynthesis) return;

    // CLEAR THE QUEUE: Stops the "traffic jam"
    window.speechSynthesis.cancel();

    const utterance = new SpeechSynthesisUtterance(text);
    
    // VOICE FALLBACK
    const availableVoices = window.speechSynthesis.getVoices();
    const preferredVoice = availableVoices.find(v => v.lang === 'en-US' && v.name.includes('Google'));
    utterance.voice = voice || preferredVoice || availableVoices[0];

    utterance.rate = 1.1;
    utterance.pitch = 0.95;
    utterance.volume = 1.0;

    utterance.onerror = (event) => {
      console.error("Squawk Error:", event.error);
    };

    window.speechSynthesis.speak(utterance);
  }, [isMuted, voice]);

  useEffect(() => {
    if (latestSignal && !isMuted) {
      if (latestSignal.clvrScore >= 8.0 || Math.abs(latestSignal.fundingRate) > 0.01) {
        triggerSquawk(latestSignal.message);
      }
    }
  }, [latestSignal, isMuted, triggerSquawk]);

  return (
    <div style={{
      padding: '10px', border: '1px solid #333', borderRadius: '8px',
      backgroundColor: isMuted ? '#1a1a1a' : '#003300', display: 'flex',
      alignItems: 'center', gap: '12px', transition: 'all 0.3s ease'
    }}>
      <button 
        onClick={() => setIsMuted(!isMuted)}
        style={{
          padding: '8px 16px', cursor: 'pointer', borderRadius: '4px', border: 'none',
          backgroundColor: isMuted ? '#ff4d4d' : '#4CAF50', color: 'white', fontWeight: 'bold'
        }}
      >
        {isMuted ? '🔈 SQUAWK OFF' : '🔊 SQUAWK LIVE'}
      </button>
      <span style={{ fontSize: '12px', color: '#ccc' }}>
        {isMuted ? 'Audio Alerts Paused' : 'Monitoring Live Feed...'}
      </span>
    </div>
  );
};

export default SquawkBox;

// --- FINNHUB + HYPERLIQUID FETCHER ---
async function fetchUniversalCandles(ticker, assetClass, interval = "1h", count = 200) {
  try {
    if (assetClass === "crypto") {
      const endTime = Date.now();
      const startTime = endTime - count * 3600000;
      const r = await fetch("https://api.hyperliquid.xyz/info", {
        method: "POST", headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ type: "candleSnapshot", req: { coin: ticker, interval, startTime, endTime } }),
      });
      const data = await r.json();
      if (!Array.isArray(data) || data.length === 0) return null;
      return data.map(c => ({ t: c.t, o: parseFloat(c.o), h: parseFloat(c.h), l: parseFloat(c.l), c: parseFloat(c.c), v: parseFloat(c.v || 0) }));
    }

    const API_KEY = process.env.FINNHUB_API_KEY || "demo"; 
    const resolution = interval === "1h" ? "60" : "D"; 
    const to = Math.floor(Date.now() / 1000); 
    const from = to - (count * 3600); 

    const r = await fetch(`https://finnhub.io/api/v1/stock/candle?symbol=${ticker}&resolution=${resolution}&from=${from}&to=${to}&token=${API_KEY}`);
    const data = await r.json();
    
    if (data.s !== "ok") return null;
    
    const formattedCandles = [];
    for (let i = 0; i < data.t.length; i++) {
      formattedCandles.push({
        t: data.t[i] * 1000, o: parseFloat(data.o[i]), h: parseFloat(data.h[i]), l: parseFloat(data.l[i]),
        c: parseFloat(data.c[i]), v: parseFloat(data.v[i] || 0)
      });
    }
    return formattedCandles;
  } catch (err) {
    console.error(`Fetch failed for ${ticker}:`, err);
    return null;
  }
}

function computeIndicators(candles) {
  if (!candles || candles.length < 50) return null;
  const closes  = candles.map(c => c.c);
  const highs   = candles.map(c => c.h);
  const lows    = candles.map(c => c.l);
  const volumes = candles.map(c => c.v);
  const n       = closes.length;

  function ema(data, period) {
    const k  = 2 / (period + 1);
    let val  = data.slice(0, period).reduce((a,b) => a+b, 0) / period;
    const out = [val];
    for (let i = period; i < data.length; i++) {
      val = data[i] * k + val * (1 - k);
      out.push(val);
    }
    return out;
  }

  const ema20  = ema(closes, 20);
  const ema50  = ema(closes, 50);
  const ema200 = ema(closes, Math.min(200, n - 1));

  const currentEma20  = ema20[ema20.length - 1];
  const currentEma50  = ema50[ema50.length - 1];
  const currentEma200 = ema200[ema200.length - 1];
  const currentPrice  = closes[n - 1];

  let gains = 0, losses = 0;
  for (let i = n - 14; i < n; i++) {
    const diff = closes[i] - closes[i-1];
    if (diff > 0) gains += diff; else losses -= diff;
  }
  const rs  = gains / Math.max(losses, 0.0001);
  const rsi = Math.round(100 - 100 / (1 + rs));

  const ema12    = ema(closes, 12);
  const ema26    = ema(closes, 26);
  const macdLine = ema12.slice(ema12.length - ema26.length).map((v, i) => v - ema26[i]);
  const signal   = ema(macdLine, 9);
  const macd     = macdLine[macdLine.length - 1];
  const macdSig  = signal[signal.length - 1];
  const macdHist = macd - macdSig;
  const prevHist = macdLine[macdLine.length - 2] - signal[signal.length - 2];

  const trValues = [];
  for (let i = Math.max(1, n - 30); i < n; i++) {
    trValues.push(Math.max(highs[i] - lows[i], Math.abs(highs[i] - closes[i-1]), Math.abs(lows[i] - closes[i-1])));
  }
  const atr14 = trValues.slice(-14).reduce((a,b) => a+b, 0) / 14;
  const atrPct = (atr14 / currentPrice) * 100;

  const recent = candles.slice(-50);
  const recentHighs = recent.map(c => c.h).sort((a,b) => b-a);
  const recentLows  = recent.map(c => c.l).sort((a,b) => a-b);
  const resistance = [recentHighs[0], recentHighs[Math.floor(recentHighs.length * 0.1)]].filter(v => v > currentPrice);
  const support = [recentLows[0], recentLows[Math.floor(recentLows.length * 0.1)]].filter(v => v < currentPrice);

  const avgVolume  = volumes.slice(-20).reduce((a,b) => a+b, 0) / 20;
  const lastVolume = volumes[n-1];
  const volumeRatio = lastVolume / avgVolume;
  const volumeSignal = volumeRatio > 2 ? "SURGE" : volumeRatio > 1.3 ? "ABOVE AVG" : "NORMAL";

  const trend = currentPrice > currentEma20 && currentEma20 > currentEma50 ? "STRONG UPTREND" : currentPrice < currentEma20 && currentEma20 < currentEma50 ? "STRONG DOWNTREND" : "RANGING";

  let momentumScore = 50;
  if (rsi > 60) momentumScore += 10;
  if (rsi < 40) momentumScore -= 10;
  if (macdHist > 0 && macdHist > prevHist) momentumScore += 10;
  momentumScore = Math.max(0, Math.min(100, momentumScore));

  return {
    ema20: currentEma20, ema50: currentEma50, ema200: currentEma200, rsi,
    rsiLabel: rsi > 70 ? "OVERBOUGHT" : rsi > 60 ? "BULLISH" : rsi < 30 ? "OVERSOLD" : "NEUTRAL",
    macd: parseFloat(macd.toFixed(4)), macdSignal: parseFloat(macdSig.toFixed(4)), macdHist: parseFloat(macdHist.toFixed(4)),
    macdCrossing: macdHist > 0 ? "BULLISH" : "BEARISH", atr14: parseFloat(atr14.toFixed(4)), atrPct: parseFloat(atrPct.toFixed(3)),
    nearestResistance: resistance[0] || null, nearestSupport: support[0] || null, volumeSignal, trend, momentumScore, currentPrice,
  };
}

function formatIndicatorsForAI(ind, profile) {
  if (!ind) return "Technical indicators unavailable.";
  const p = profile.label;
  return `TECHNICAL ANALYSIS — ${p} PROFILE:
  Trend Structure: ${ind.trend}
  Momentum Score: ${ind.momentumScore}/100
  EMA20: $${fmtPrice(ind.ema20)} | EMA50: $${fmtPrice(ind.ema50)} | EMA200: $${fmtPrice(ind.ema200)}
  RSI(14): ${ind.rsi} → ${ind.rsiLabel}
  MACD Hist: ${ind.macdHist} → ${ind.macdCrossing}
  ATR(14): $${ind.atr14} (${ind.atrPct}% of price)
  Nearest Resistance: $${fmtPrice(ind.nearestResistance) || "none"} | Nearest Support: $${fmtPrice(ind.nearestSupport) || "none"}
  Volume Signal: ${ind.volumeSignal}`;
}

async function runQuantAnalysis({ ticker, assetClass, profile, liveData, indicators, twitterContext }) {
  const { perpPx, spotPx } = liveData;
  const price = assetClass === "crypto" ? perpPx(ticker) : (spotPx(ticker) || perpPx(ticker) || indicators.currentPrice);

  if (!price || price === 0) throw new Error(`No live price for ${ticker} — analysis blocked`);

  const system = `You are an elite quantitative analyst. Respond ONLY with valid JSON matching the exact requested schema.`;
  const user = `QUANT ANALYSIS REQUEST
  Asset: ${ticker} | Class: ${assetClass.toUpperCase()} | Profile: ${profile.label}
  Current Price: $${fmtPrice(price)}
  ${formatIndicatorsForAI(indicators, profile)}
  Please provide JSON containing signal, entry, stopLoss, tp1, tp2, leverage, hold, technical_summary, quant_rationale, risks, invalidation, and opportunity_score.`;

  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST", headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ model: "claude-sonnet-4-20250514", max_tokens: 2000, system, messages: [{ role:"user", content:user }] }),
  });

  const data = await res.json();
  const raw = data.content?.map(b => b.text || "").join("") || "";
  const clean = raw.replace(/`json|`/g, "").trim();
  const parsed = JSON.parse(clean);

  parsed.ticker = ticker;
  parsed.generatedAt = new Date().toLocaleTimeString();
  return parsed;
}
