### 2C. Outcome Resolver Worker

Create a background worker that resolves PENDING signals every 60 seconds:

```typescript
// server/lib/outcomeResolver.ts (NEW FILE)
import { db } from '../db';
import { ai_signal_log } from '../schema';
import { eq, and, lte } from 'drizzle-orm';

// You'll need access to current prices — import from your existing market data source
// e.g., from DataBus prices, or Hyperliquid WebSocket store, or Finnhub

async function getCurrentPrice(token: string): Promise<number | null> {
  // Use your existing price fetching mechanism
  // Check DataBus prices first (fastest), then Finnhub/CoinGecko as fallback
  try {
    const response = await fetch(`http://localhost:${process.env.PORT || 3000}/api/databus/status`);
    const data = await response.json();
    return data.prices?.[token] || null;
  } catch {
    return null;
  }
}

export async function resolveSignals() {
  try {
    // Get all PENDING signals
    const pending = await db.select().from(ai_signal_log)
      .where(eq(ai_signal_log.outcome, 'PENDING'));

    const now = new Date();

    for (const signal of pending) {
      const price = await getCurrentPrice(signal.token);
      if (!price || !signal.entry_price) continue;

      const entry = parseFloat(signal.entry_price.toString());
      const isLong = signal.direction === 'LONG';

      // Check SL hit first (highest priority)
      if (signal.stop_loss) {
        const sl = parseFloat(signal.stop_loss.toString());
        const slHit = isLong ? price <= sl : price >= sl;
        if (slHit) {
          const pnl = isLong ? ((sl - entry) / entry) * 100 : ((entry - sl) / entry) * 100;
          await updateOutcome(signal.id, 'SL_HIT', pnl);
          continue;
        }
      }

      // Check TP3 (best outcome)
      if (signal.tp3_price) {
        const tp3 = parseFloat(signal.tp3_price.toString());
        const tp3Hit = isLong ? price >= tp3 : price <= tp3;
        if (tp3Hit) {
          const pnl = isLong ? ((tp3 - entry) / entry) * 100 : ((entry - tp3) / entry) * 100;
          await updateOutcome(signal.id, 'TP3_HIT', pnl);
          continue;
        }
      }

      // Check TP2
      if (signal.tp2_price) {
        const tp2 = parseFloat(signal.tp2_price.toString());
        const tp2Hit = isLong ? price >= tp2 : price <= tp2;
        if (tp2Hit) {
          const pnl = isLong ? ((tp2 - entry) / entry) * 100 : ((entry - tp2) / entry) * 100;
          await updateOutcome(signal.id, 'TP2_HIT', pnl);
          continue;
        }
      }

      // Check TP1
      if (signal.tp1_price) {
        const tp1 = parseFloat(signal.tp1_price.toString());
        const tp1Hit = isLong ? price >= tp1 : price <= tp1;
        if (tp1Hit) {
          const pnl = isLong ? ((tp1 - entry) / entry) * 100 : ((entry - tp1) / entry) * 100;
          await updateOutcome(signal.id, 'TP1_HIT', pnl);
          continue;
        }
      }

      // Check kill clock expiry
      if (signal.kill_clock_expires && now >= new Date(signal.kill_clock_expires)) {
        const pnl = isLong ? ((price - entry) / entry) * 100 : ((entry - price) / entry) * 100;
        const outcome = pnl >= 0 ? 'EXPIRED_WIN' : 'EXPIRED_LOSS';
        await updateOutcome(signal.id, outcome, pnl);
        continue;
      }
    }
  } catch (err) {
    console.error('[OutcomeResolver] Error:', err);
  }
}

async function updateOutcome(id: number, outcome: string, pnl: number) {
  await db.update(ai_signal_log)
    .set({
      outcome,
      pnl_pct: Math.round(pnl * 100) / 100,
      resolved_at: new Date(),
    })
    .where(eq(ai_signal_log.id, id));

  console.log(`[OutcomeResolver] Signal #${id} resolved: ${outcome} (${pnl >= 0 ? '+' : ''}${pnl.toFixed(2)}%)`);
}

// Start the resolver — call this from server startup
export function startOutcomeResolver() {
  console.log('[OutcomeResolver] Started — checking every 60s');
  resolveSignals(); // Run immediately on startup
  setInterval(resolveSignals, 60000); // Then every 60 seconds
}
```

### 3A. Performance Context for AI Prompts

Create a function that builds a performance context string from historical outcomes:

```typescript
// server/lib/performanceContext.ts (NEW FILE)

export async function buildPerformanceContext(): Promise<string> {
  try {
    // Query last 30 days of resolved signals from ai_signal_log
    const thirtyDaysAgo = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);

    const resolved = await db.select().from(ai_signal_log)
      .where(and(
        sql`${ai_signal_log.outcome} != 'PENDING'`,
        sql`${ai_signal_log.created_at} >= ${thirtyDaysAgo}`
      ));

    if (resolved.length < 5) {
      return 'HISTORICAL PERFORMANCE: Insufficient data (<5 resolved signals in last 30 days). Use standard parameters.';
    }

    const wins = resolved.filter(s => ['TP1_HIT','TP2_HIT','TP3_HIT','EXPIRED_WIN'].includes(s.outcome));
    const losses = resolved.filter(s => ['SL_HIT','EXPIRED_LOSS'].includes(s.outcome));
    const overallWinRate = Math.round((wins.length / resolved.length) * 100);

    // By asset
    const byAsset = {};
    resolved.forEach(s => {
      if (!byAsset[s.token]) byAsset[s.token] = { wins: 0, total: 0, pnls: [] };
      byAsset[s.token].total++;
      if (['TP1_HIT','TP2_HIT','TP3_HIT','EXPIRED_WIN'].includes(s.outcome)) byAsset[s.token].wins++;
      if (s.pnl_pct) byAsset[s.token].pnls.push(parseFloat(s.pnl_pct));
    });

    // By direction
    const byDirection = {};
    resolved.forEach(s => {
      if (!byDirection[s.direction]) byDirection[s.direction] = { wins: 0, total: 0 };
      byDirection[s.direction].total++;
      if (['TP1_HIT','TP2_HIT','TP3_HIT','EXPIRED_WIN'].includes(s.outcome)) byDirection[s.direction].wins++;
    });

    // By trade type
    const byTradeType = {};
    resolved.forEach(s => {
      const tt = s.trade_type || 'UNKNOWN';
      if (!byTradeType[tt]) byTradeType[tt] = { wins: 0, total: 0 };
      byTradeType[tt].total++;
      if (['TP1_HIT','TP2_HIT','TP3_HIT','EXPIRED_WIN'].includes(s.outcome)) byTradeType[tt].wins++;
    });

    // Recent losing trades (last 5)
    const recentLosses = losses.slice(-5).map(s =>
      `${s.token} ${s.direction} (${s.trade_type || '?'}) — ${s.outcome}, ${s.pnl_pct}%`
    );

    // Recent winning trades (last 5)
    const recentWins = wins.slice(-5).map(s =>
      `${s.token} ${s.direction} (${s.trade_type || '?'}) — ${s.outcome}, +${s.pnl_pct}%`
    );

    // Build the context string
    let ctx = `HISTORICAL PERFORMANCE CONTEXT (last 30 days, ${resolved.length} resolved signals):\n`;
    ctx += `Overall win rate: ${overallWinRate}% (${wins.length}W / ${losses.length}L)\n\n`;

    ctx += `By asset:\n`;
    Object.entries(byAsset).forEach(([token, data]) => {
      const wr = Math.round((data.wins / data.total) * 100);
      const avgPnl = data.pnls.length ? (data.pnls.reduce((a,b) => a+b, 0) / data.pnls.length).toFixed(2) : '?';
      ctx += `  ${token}: ${wr}% win rate (${data.total} signals, avg PnL ${avgPnl}%)\n`;
    });

    ctx += `\nBy direction:\n`;
    Object.entries(byDirection).forEach(([dir, data]) => {
      ctx += `  ${dir}: ${Math.round((data.wins / data.total) * 100)}% win rate (${data.total} signals)\n`;
    });

    ctx += `\nBy trade type:\n`;
    Object.entries(byTradeType).forEach(([tt, data]) => {
      ctx += `  ${tt}: ${Math.round((data.wins / data.total) * 100)}% win rate (${data.total} signals)\n`;
    });

    if (recentLosses.length > 0) {
      ctx += `\nRecent losses (AVOID similar setups):\n`;
      recentLosses.forEach(l => { ctx += `  ❌ ${l}\n`; });
    }

    if (recentWins.length > 0) {
      ctx += `\nRecent wins (FAVOR similar setups):\n`;
      recentWins.forEach(w => { ctx += `  ✅ ${w}\n`; });
    }

    ctx += `\nINSTRUCTIONS: Use this performance data to improve signal quality. Favor asset/direction/trade-type combos with >65% win rate. Be MORE cautious with combos below 50%. If a specific asset has been consistently losing, either skip it or require higher conviction (75%+).`;

    return ctx;
  } catch (err) {
    console.error('[PerformanceContext] Error building context:', err);
    return 'HISTORICAL PERFORMANCE: Error loading performance data. Use standard parameters.';
  }
}
```

**Threshold recalculation job (runs every hour):**

```typescript
// server/lib/adaptiveThresholds.ts (NEW FILE)

export async function recalculateThresholds() {
  console.log('[AdaptiveThresholds] Recalculating...');

  const thirtyDaysAgo = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);

  // Get all resolved signals in last 30 days, grouped by token + direction
  const resolved = await db.select().from(ai_signal_log)
    .where(and(
      sql`${ai_signal_log.outcome} != 'PENDING'`,
      sql`${ai_signal_log.created_at} >= ${thirtyDaysAgo}`
    ));

  // Group by token + direction
  const groups = {};
  resolved.forEach(s => {
    const key = `${s.token}__${s.direction}`;
    if (!groups[key]) groups[key] = { token: s.token, direction: s.direction, wins: 0, total: 0 };
    groups[key].total++;
    if (['TP1_HIT','TP2_HIT','TP3_HIT','EXPIRED_WIN'].includes(s.outcome)) groups[key].wins++;
  });

  for (const [key, data] of Object.entries(groups)) {
    if (data.total < 5) continue; // Not enough data to adjust

    const winRate = (data.wins / data.total) * 100;
    let adjustment = 0;

    if (winRate < 40) {
      adjustment = 15;        // Very bad — significantly raise threshold
    } else if (winRate < 50) {
      adjustment = 10;        // Bad — raise threshold
    } else if (winRate < 55) {
      adjustment = 5;         // Below average — slight raise
    } else if (winRate >= 55 && winRate <= 65) {
      adjustment = 0;         // Acceptable — no change
    } else if (winRate > 65 && winRate <= 75) {
      adjustment = -5;        // Good — lower threshold slightly
    } else if (winRate > 75) {
      adjustment = -10;       // Excellent — lower threshold more
    }

    // Clamp to ±20
    adjustment = Math.max(-20, Math.min(20, adjustment));

    const suppressed = winRate < 40 && data.total >= 10;

    // Upsert threshold
    await db.insert(adaptive_thresholds).values({
      token: data.token,
      direction: data.direction,
      trade_type: 'ALL',
      baseline_threshold: 75,
      current_threshold: 75 + adjustment,
      adjustment,
      win_rate_30d: Math.round(winRate * 100) / 100,
      sample_size: data.total,
      suppressed,
      last_recalc: new Date(),
      updated_at: new Date(),
    }).onConflictDoUpdate({
      target: [adaptive_thresholds.token, adaptive_thresholds.direction, adaptive_thresholds.trade_type],
      set: {
        current_threshold: 75 + adjustment,
        adjustment,
        win_rate_30d: Math.round(winRate * 100) / 100,
        sample_size: data.total,
        suppressed,
        last_recalc: new Date(),
        updated_at: new Date(),
      },
      // Don't override manual overrides:
      where: sql`${adaptive_thresholds.manual_override} = false`,
    });

    console.log(`[AdaptiveThresholds] ${data.token} ${data.direction}: ${winRate.toFixed(0)}% win rate → threshold ${75 + adjustment}% (adj ${adjustment >= 0 ? '+' : ''}${adjustment})`);
  }
}

// Start hourly recalculation
export function startAdaptiveThresholds() {
  console.log('[AdaptiveThresholds] Started — recalculating every hour');
  recalculateThresholds();
  setInterval(recalculateThresholds, 3600000); // Every hour
}
```

```javascript
    const handle = () => setWidth(window.innerWidth);
```
