```typescript
export async function getNewsImpact(
  token: string,
  direction: "LONG" | "SHORT",
  lookbackMinutes: number = 240
): Promise<NewsImpact> {
  const keywords = ASSET_NEWS_MAP[token.toUpperCase()] || [token.toUpperCase()];
  const cutoff = new Date(Date.now() - lookbackMinutes * 60 * 1000);

  const recent = await db
    .select()
    .from(newsItems)
    .where(gte(newsItems.createdAt, cutoff))
    .orderBy(desc(newsItems.createdAt))
    .limit(50);

  const relevant = recent.filter((item) => {
    const text = `${item.title} ${item.tickers || ""}`.toUpperCase();
    return keywords.some((kw) => text.includes(kw));
  });

  let bearish = 0, bullish = 0, neutral = 0;
  let maxSeverity: "low" | "medium" | "high" | null = null;
  const headlines: string[] = [];

  for (const item of relevant) {
    const sentiment = (item.sentiment || "").toLowerCase();
    const severity = (item.severity || "").toLowerCase();

    if (sentiment === "bearish") bearish++;
    else if (sentiment === "bullish") bullish++;
    else neutral++;

    if (severity === "high") maxSeverity = "high";
    else if (severity === "medium" && maxSeverity !== "high") maxSeverity = "medium";
    else if (severity === "low" && !maxSeverity) maxSeverity = "low";

    if (headlines.length < 3) headlines.push(item.title);
  }

  const contradicts =
    (direction === "LONG" && bearish > bullish) ||
    (direction === "SHORT" && bullish > bearish);

  let confidenceAdjustment = 0;
  let shouldBlock = false;

  if (contradicts && maxSeverity === "high") {
    shouldBlock = true;
    confidenceAdjustment = -100;
  } else if (contradicts && maxSeverity === "medium") {
    confidenceAdjustment = -20;
  } else if (contradicts) {
    confidenceAdjustment = -10;
  } else if (!contradicts && maxSeverity === "high") {
    confidenceAdjustment = +5;
  }

  return {
    hasConflict: contradicts,
    severity: maxSeverity,
    bearishCount: bearish,
    bullishCount: bullish,
    neutralCount: neutral,
    topHeadlines: headlines,
    confidenceAdjustment,
    shouldBlock,
  };
}
```

### 2A. Tier resolver

```typescript
import { db } from "../db";
import { users } from "../../shared/schema";
import { eq } from "drizzle-orm";

export type Tier = "free" | "pro" | "elite";

export async function getUserTier(userId: string): Promise<Tier> {
  const [user] = await db.select().from(users).where(eq(users.id, userId));
  if (!user) return "free";

  // Adjust field name to match your schema (subscriptionTier, tier, plan, etc.)
  const tier = (user.subscriptionTier || user.tier || "free").toLowerCase();

  if (tier === "elite" || tier === "premium") return "elite";
  if (tier === "pro" || tier === "standard") return "pro";
  return "free";
}

export function getMaxIdeasForTier(tier: Tier): number {
  return tier === "elite" ? 3 : tier === "pro" ? 1 : 0;
}
```

```typescript
import { getUserTier } from "./lib/userTier";
import { userPromotedAssets } from "../shared/schema";

const MAX_PROMOTED_ASSETS = 5;

// GET user's promoted assets
app.get("/api/basket/promoted", requireAuth, async (req, res) => {
  const userId = req.user.id;
  const promoted = await db
    .select()
    .from(userPromotedAssets)
    .where(eq(userPromotedAssets.userId, userId));
  res.json(promoted);
});

// POST promote an asset
app.post("/api/basket/promote", requireAuth, async (req, res) => {
  const userId = req.user.id;
  const tier = await getUserTier(userId);

  if (tier !== "elite") {
    return res.status(403).json({ error: "Promote-to-scanner is Elite-only" });
  }

  const { assetSymbol, assetClass, yahooSymbol } = req.body;
  if (!assetSymbol || !assetClass || !yahooSymbol) {
    return res.status(400).json({ error: "Missing fields" });
  }

  const existing = await db
    .select()
    .from(userPromotedAssets)
    .where(eq(userPromotedAssets.userId, userId));

  if (existing.length >= MAX_PROMOTED_ASSETS) {
    return res.status(400).json({ error: `Max ${MAX_PROMOTED_ASSETS} promoted assets. Remove one first.` });
  }
  if (existing.some((e) => e.assetSymbol === assetSymbol)) {
    return res.status(400).json({ error: "Already promoted" });
  }

  const [inserted] = await db
    .insert(userPromotedAssets)
    .values({ userId, assetSymbol, assetClass, yahooSymbol })
    .returning();

  res.json(inserted);
});

// DELETE demote an asset
app.delete("/api/basket/promote/:id", requireAuth, async (req, res) => {
  const userId = req.user.id;
  const id = parseInt(req.params.id);

  await db
    .delete(userPromotedAssets)
    .where(and(eq(userPromotedAssets.id, id), eq(userPromotedAssets.userId, userId)));

  res.json({ ok: true });
});
```
