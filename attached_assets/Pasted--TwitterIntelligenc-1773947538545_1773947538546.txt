// API HELPER

async function twitterGet(endpoint, params = {}) {
if (!RAPIDAPI_KEY) return null;
try {
const url = new URL(`${BASE_URL}${endpoint}`);
Object.entries(params).forEach(([k, v]) => url.searchParams.set(k, v));
const r = await fetch(url.toString(), { headers: HEADERS });
if (!r.ok) return null;
return await r.json();
} catch { return null; }
}

// Safely extract tweets from any Twitter135 response shape
function extractTweets(data) {
return (
data?.result?.timeline?.instructions
?.flatMap(i => i.entries || [])
?.filter(e => e.content?.itemContent?.tweet_results?.result)
?.map(e => {
const t    = e.content.itemContent.tweet_results.result;
const user = t.core?.user_results?.result?.legacy;
return {
id:        t.legacy?.id_str || “”,
text:      t.legacy?.full_text || “”,
likes:     t.legacy?.favorite_count || 0,
retweets:  t.legacy?.retweet_count  || 0,
createdAt: t.legacy?.created_at     || “”,
author:    user?.name               || “Unknown”,
handle:    user?.screen_name        || “”,
};
}) || []
);
}

const BULLISH_WORDS = [

“bear”,“bearish”,“dump”,“crash”,“sell”,“short”,“collapse”,“drop”,“fall”,
“rug”,“scam”,“dead”,“rekt”,“liquidated”,“📉”,“🔴”,“🩸”,“down”,

function classifySentiment(text) {
if (!text) return “neutral”;
const t = text.toLowerCase();
let bull = 0, bear = 0;
BULLISH_WORDS.forEach(w => { if (t.includes(w)) bull++; });
BEARISH_WORDS.forEach(w => { if (t.includes(w)) bear++; });
if (bull === 0 && bear === 0) return “neutral”;
if (bull > bear) return “bullish”;
if (bear > bull) return “bearish”;
return “neutral”;
}

function isWithinHours(dateStr, hours) {
if (!dateStr) return false;
const ts = new Date(dateStr).getTime();
return !isNaN(ts) && (Date.now() - ts) < hours * 3_600_000;
}

export function timeAgo(dateStr) {
if (!dateStr) return “”;
const diff = Date.now() - new Date(dateStr).getTime();
const m    = Math.floor(diff / 60000);
if (m < 1)  return “just now”;
if (m < 60) return `${m}m ago`;
const h = Math.floor(m / 60);
if (h < 24) return `${h}h ago`;
return `${Math.floor(h / 24)}d ago`;
}

// STREAM 4 — Overall sentiment score
// ─────────────────────────────────────────────────────────────────────────────
async function fetchOverallSentiment() {
const data   = await twitterGet(”/SearchV2/”, {
query: “crypto market stocks bitcoin ethereum lang:en -is:retweet”,
count: “40”,
type:  “Latest”,
});
const tweets = extractTweets(data).filter(t => isWithinHours(t.createdAt, 2));
const total  = Math.max(tweets.length, 1);
const bull   = tweets.filter(t => classifySentiment(t.text) === “bullish”).length;
const bear   = tweets.filter(t => classifySentiment(t.text) === “bearish”).length;
const score  = Math.round((bull / total) * 100);

// Weighted by engagement — viral tweets count more
const wBull = tweets.filter(t => classifySentiment(t.text) === “bullish”)
.reduce((s, t) => s + 1 + Math.log(t.likes + 1), 0);
const wBear = tweets.filter(t => classifySentiment(t.text) === “bearish”)
.reduce((s, t) => s + 1 + Math.log(t.likes + 1), 0);
const wTotal = wBull + wBear + 0.001;

return {
score,
weightedScore: Math.round((wBull / wTotal) * 100),
label:
score > 65 ? “VERY BULLISH” :
score > 55 ? “BULLISH”      :
score < 35 ? “VERY BEARISH” :
score < 45 ? “BEARISH”      : “NEUTRAL”,
bullishCount:  bull,
bearishCount:  bear,
totalTweets:   tweets.length,
sampleSize:    `${tweets.length} tweets (last 2h)`,
};
}

// MASTER STORE — module-level singleton, 4-minute refresh
// ─────────────────────────────────────────────────────────────────────────────
const STORE = {
data:      null,
fetching:  false,
lastFetch: 0,
listeners: new Set(),

isStale()     { return !this.lastFetch || (Date.now() - this.lastFetch) > REFRESH_INTERVAL_MS; },
subscribe(fn) { this.listeners.add(fn); return () => this.listeners.delete(fn); },
notify()      { this.listeners.forEach(fn => fn(this.data)); },

async refresh() {
if (this.fetching || !RAPIDAPI_KEY) return;
this.fetching = true;
try {
// Fetch streams in two batches — slight stagger avoids rate limit
const [whales, sentiment] = await Promise.all([
fetchWhaleActivity(),
fetchOverallSentiment(),
]);
await new Promise(r => setTimeout(r, 600));
const [mentions, breaking] = await Promise.all([
fetchTickerMentions(),
fetchBreakingTweets(),
]);

```
  this.data = {
    whales, mentions, breaking, sentiment,
    fetchedAt:    Date.now(),
    fetchedAtStr: new Date().toLocaleTimeString(),
  };
  this.lastFetch = Date.now();
  this.notify();
} catch (e) {
  console.error("[TwitterIntelligence]", e);
} finally {
  this.fetching = false;
}
```

},
};

// Auto-refresh loop — starts once globally
let _started = false;
function startLoop() {
if (_started || !RAPIDAPI_KEY) return;
_started = true;
STORE.refresh();
setInterval(() => { if (STORE.isStale()) STORE.refresh(); }, 60_000);
}

export function useTwitterIntelligence() {
const [data,    setData]    = useState(STORE.data);
const [loading, setLoading] = useState(!STORE.data);

useEffect(() => {
startLoop();
if (STORE.data) { setData(STORE.data); setLoading(false); }
const unsub = STORE.subscribe(d => { setData(d); setLoading(false); });
return unsub;
}, []);

return {
whales:    data?.whales    || [],
mentions:  data?.mentions  || {},
breaking:  data?.breaking  || [],
sentiment: data?.sentiment || { score:50, label:“NEUTRAL”, totalTweets:0 },
loading,
hasKey:       !!RAPIDAPI_KEY,
hasData:      !!data,
fetchedAtStr: data?.fetchedAtStr || “”,
timeAgo,
refresh:      () => STORE.refresh(),
// Pre-built AI context string — inject directly into AI prompts
aiContext:    data ? buildTwitterContext(data) : “”,
};
}

// Full context — use in Morning Brief and general AI analysis
export function buildTwitterContext(data) {
if (!data) return “Twitter/X data unavailable.”;
const { whales, mentions, breaking, sentiment } = data;
const lines = [“── TWITTER/X SOCIAL INTELLIGENCE ──”];

lines.push(`Overall Sentiment: ${sentiment.label} (${sentiment.score}% bullish, weighted: ${sentiment.weightedScore}%)`);
lines.push(`Sample: ${sentiment.sampleSize}`);
lines.push(””);

const recentWhales = whales.filter(w => w.isRecent);
if (recentWhales.length) {
lines.push(“WHALE ACTIVITY (last 4h):”);
recentWhales.slice(0, 4).forEach(w => {
lines.push(`  @${w.whale.handle}${w.isViral?" [VIRAL]":""}: "${w.tweet.text.slice(0,120)}"`);
lines.push(`  → ${w.sentiment.toUpperCase()} | ${w.tweet.likes} likes${w.assets.length?` | ${w.assets.join(”,”)}`:""}`);
});
lines.push(””);
}

const spikes = Object.values(mentions).filter(m => m.isSpiking);
if (spikes.length) {
lines.push(“MENTION SPIKES (unusual 1h activity):”);
spikes.forEach(m => lines.push(`  ${m.ticker}: ${m.count1h} mentions | ${m.sentimentScore}% bullish`));
lines.push(””);
}

const normal = Object.values(mentions).filter(m => !m.isSpiking && m.count1h > 0);
if (normal.length) {
lines.push(“TICKER MENTIONS (last 1h):”);
normal.forEach(m => lines.push(`  ${m.ticker}: ${m.count1h} tweets | ${m.sentimentScore}% bullish`));
lines.push(””);
}

if (breaking.length) {
lines.push(“BREAKING / VIRAL TWEETS:”);
breaking.slice(0, 3).forEach(t => {
lines.push(`  @${t.handle}: "${t.text.slice(0,100)}"`);
lines.push(`  → ${t.sentiment.toUpperCase()} | ${t.likes} likes | ${t.assets.join(",")||"general"}`);
});
}

lines.push(`[Fetched: ${data.fetchedAtStr} | Refreshes every 4 min | ~43,200 req/mo]`);
lines.push(“── END TWITTER INTELLIGENCE ──”);
return lines.join(”\n”);
}

// Asset-specific context — use in Signals Tab AI prompt
export function buildAssetTwitterContext(data, ticker) {
if (!data) return `No Twitter data available for ${ticker}.`;
const { whales, mentions, breaking, sentiment } = data;
const lines = [];

const tickerData = mentions[`$${ticker}`] || mentions[ticker];
if (tickerData) {
lines.push(`${ticker} Twitter (last 1h): ${tickerData.count1h} mentions | ${tickerData.sentimentScore}% bullish | Bullish:${tickerData.bullishCount} Bearish:${tickerData.bearishCount}`);
if (tickerData.isSpiking) lines.push(`⚠ MENTION SPIKE — unusually high ${ticker} activity`);
}

const relWhales = whales.filter(w => w.assets?.includes(ticker) && w.isRecent);
relWhales.slice(0, 2).forEach(w =>
lines.push(`@${w.whale.handle}: "${w.tweet.text.slice(0,100)}" → ${w.sentiment.toUpperCase()} (${w.tweet.likes} likes)`)
);

const relBreaking = breaking.filter(t => t.assets?.includes(ticker));
relBreaking.slice(0, 2).forEach(t =>
lines.push(`Breaking @${t.handle}: "${t.text.slice(0,100)}" → ${t.sentiment.toUpperCase()}`)
);

lines.push(`Overall X mood: ${sentiment.label} (${sentiment.score}%)`);
return lines.join(”\n”) || `No specific ${ticker} Twitter activity in last 2h.`;
}

// Morning brief builder — structured data for Morning Brief Tab
export function buildTwitterMorningSummary(data) {
if (!data) return null;
const { whales, mentions, breaking, sentiment } = data;
const topAsset     = Object.values(mentions).sort((a,b) => b.count1h - a.count1h)[0];
const mostBullish  = Object.values(mentions).filter(m=>m.count1h>3).sort((a,b)=>b.sentimentScore-a.sentimentScore)[0];
const mostBearish  = Object.values(mentions).filter(m=>m.count1h>3).sort((a,b)=>a.sentimentScore-b.sentimentScore)[0];
const viralWhales  = whales.filter(w => w.isViral && w.isRecent);
const trend        = sentiment.score > 60 ? “IMPROVING” : sentiment.score < 40 ? “DETERIORATING” : “STABLE”;
return {
overallScore:     sentiment.score,
overallLabel:     sentiment.label,
sentimentTrend:   trend,
topMentionedAsset:topAsset?.ticker,
mostBullishAsset: mostBullish?.ticker,
mostBearishAsset: mostBearish?.ticker,
viralWhaleCount:  viralWhales.length,
viralWhales:      viralWhales.slice(0, 2),
breakingCount:    breaking.length,
topBreaking:      breaking[0] || null,
tickerSentiments: Object.fromEntries(Object.entries(mentions).map(([k,v]) => [k, v.sentimentScore])),
fetchedAt:        data.fetchedAtStr,
};
}
